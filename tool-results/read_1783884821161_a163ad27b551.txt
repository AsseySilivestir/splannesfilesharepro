   160→    @Override
   161→    public Response serve(IHTTPSession session) {
   162→        String uri = session.getUri();
   163→        Method method = session.getMethod();
   164→        Log.d(TAG, method + " " + uri);
   165→
   166→        if (method == Method.OPTIONS) {
   167→            return cors(newFixedLengthResponse(Response.Status.OK, MIME_HTML, ""));
   168→        }
   169→
   170→        try {
   171→            if (uri.equals("/") || uri.equals("/index.html")) {
   172→                return cors(newFixedLengthResponse(Response.Status.OK, MIME_HTML, buildWebUI()));
   173→            }
   174→            switch (uri) {
   175→                case "/api/files":       return serveFileList();
   176→                case "/api/folders":     return serveFolderList();
   177→                case "/api/folder-contents": return serveFolderContents(session);
   178→                case "/api/upload":      return serveFileUpload(session);
   179→                case "/screen":          return cors(newFixedLengthResponse(Response.Status.OK, MIME_HTML, buildScreenPage()));
   180→                case "/api/screen/frame":  return serveScreenFrame();
   181→                case "/api/screen/status": return serveScreenStatus();
   182→                case "/api/transfers":   return serveTransferLog();
   183→                case "/api/received":   return serveReceivedList();
   184→                default:
   185→                    if (uri.startsWith("/download/file/"))  return serveFileDownload(uri);
   186→                    if (uri.startsWith("/download/folder/")) return serveFolderDownload(uri);
   187→                    if (uri.startsWith("/download/received/")) return serveReceivedDownload(uri);
   188→                    return cors(newFixedLengthResponse(Response.Status.NOT_FOUND, MIME_HTML, "Not Found"));
   189→            }
   190→        } catch (Exception e) {
   191→            Log.e(TAG, "Error serving " + uri, e);
   192→            return cors(newFixedLengthResponse(Response.Status.INTERNAL_ERROR, MIME_PLAINTEXT, "Server error"));
   193→        }
   194→    }

   224→    private Response serveFolderContents(IHTTPSession session) {
   225→        Map<String, String> params = session.getParms();
   226→        String idStr = params.get("id");
   227→        String subPath = params.get("path");
   228→        int id;
   229→        try { id = Integer.parseInt(idStr); } catch (Exception e) {
   230→            return cors(newFixedLengthResponse(Response.Status.BAD_REQUEST, "application/json", "{\"error\":\"invalid id\"}"));
   231→        }
   232→        FileItem[] folders = sharedFolders.toArray(new FileItem[0]);
   233→        if (id < 0 || id >= folders.length) {
   234→            return cors(newFixedLengthResponse(Response.Status.NOT_FOUND, "application/json", "{\"error\":\"folder not found\"}"));
   235→        }
   236→        File folder = new File(folders[id].path);
   237→        if (subPath != null && !subPath.isEmpty()) folder = new File(folder, subPath);
   238→        if (!folder.exists() || !folder.isDirectory()) {
   239→            return cors(newFixedLengthResponse(Response.Status.NOT_FOUND, "application/json", "{\"error\":\"not a directory\"}"));
   240→        }
   241→        StringBuilder json = new StringBuilder("{\"path\":\"" + esc(subPath != null ? subPath : "") + "\",\"items\":[");
   242→        File[] files = folder.listFiles();
   243→        if (files != null) {
   244→            boolean first = true;
   245→            for (File file : files) {
   246→                if (!first) json.append(",");
   247→                first = false;
   248→                json.append(String.format(Locale.US, "{\"name\":\"%s\",\"size\":%d,\"isDir\":%b}",
   249→                        esc(file.getName()), file.length(), file.isDirectory()));
   250→            }
   251→        }
   252→        json.append("]}");
   253→        return cors(newFixedLengthResponse(Response.Status.OK, "application/json", json.toString()));
   254→    }

   256→    private Response serveFileDownload(String uri) {
   257→        String idStr = uri.substring("/download/file/".length());
   258→        int id;
   259→        try { id = Integer.parseInt(idStr); } catch (Exception e) {
   260→            return cors(newFixedLengthResponse(Response.Status.BAD_REQUEST, MIME_PLAINTEXT, "Invalid id"));
   261→        }
   262→        FileItem[] files = sharedFiles.toArray(new FileItem[0]);
   263→        if (id < 0 || id >= files.length) {
   264→            return cors(newFixedLengthResponse(Response.Status.NOT_FOUND, MIME_PLAINTEXT, "File not found"));
   265→        }
   266→        File file = new File(files[id].path);
   267→        if (!file.exists()) {
   268→            return cors(newFixedLengthResponse(Response.Status.NOT_FOUND, MIME_PLAINTEXT, "File not found"));
   269→        }
   270→        logTransfer("Downloaded: " + file.getName());
   271→        try {
   272→            FileInputStream fis = new FileInputStream(file);
   273→            Response r = newChunkedResponse(Response.Status.OK, getMimeType(file.getName()), fis);
   274→            addCorsHeaders(r);
   275→            return r;
   276→        } catch (IOException e) {
   277→            return cors(newFixedLengthResponse(Response.Status.INTERNAL_ERROR, MIME_PLAINTEXT, "Read error"));
   278→        }
   279→    }

   281→    private Response serveFolderDownload(String uri) {
   282→        String idStr = uri.substring("/download/folder/".length());
   283→        int id;
   284→        try { id = Integer.parseInt(idStr); } catch (Exception e) {
   285→            return cors(newFixedLengthResponse(Response.Status.BAD_REQUEST, MIME_PLAINTEXT, "Invalid id"));
   286→        }
   287→        FileItem[] folders = sharedFolders.toArray(new FileItem[0]);
   288→        if (id < 0 || id >= folders.length) {
   289→            return cors(newFixedLengthResponse(Response.Status.NOT_FOUND, MIME_PLAINTEXT, "Folder not found"));
   290→        }
   291→        File folder = new File(folders[id].path);
   292→        if (!folder.exists()) {
   293→            return cors(newFixedLengthResponse(Response.Status.NOT_FOUND, MIME_PLAINTEXT, "Folder not found"));
   294→        }
   295→        try {
   296→            File zipFile = new File(context.getCacheDir(), "share_" + System.currentTimeMillis() + ".zip");
   297→            zipFolder(folder, zipFile);
   298→            logTransfer("Downloaded folder: " + folder.getName());
   299→            FileInputStream fis = new FileInputStream(zipFile);
   300→            Response r = newChunkedResponse(Response.Status.OK, "application/zip", fis);
   301→            addCorsHeaders(r);
   302→            return r;
   303→        } catch (Exception e) {
   304→            Log.e(TAG, "Zip error", e);
   305→            return cors(newFixedLengthResponse(Response.Status.INTERNAL_ERROR, MIME_PLAINTEXT, "Zip error"));
   306→        }
   307→    }

   309→    private Response serveFileUpload(IHTTPSession session) {
   310→        try {
   311→            Map<String, String> files = new HashMap<>();
   312→            session.parseBody(files);
   313→            String tempPath = files.get("file");
   314→            if (tempPath == null) {
   315→                return cors(newFixedLengthResponse(Response.Status.BAD_REQUEST, "application/json", "{\"error\":\"no file\"}"));
   316→            }
   317→            File tempFile = new File(tempPath);
   318→            if (!tempFile.exists()) {
   319→                return cors(newFixedLengthResponse(Response.Status.BAD_REQUEST, "application/json", "{\"error\":\"upload failed\"}"));
   320→            }
   321→
   322→            String originalName = tempFile.getName();
   323→            // Clean up nanohttpd prefix if present
   324→            if (originalName.startsWith("nanohttpd-")) {
   325→                Map<String, String> headers = session.getHeaders();
   326→                String cd = headers.get("content-disposition");
   327→                if (cd != null) {
   328→                    int fi = cd.indexOf("filename=\"");
   329→                    if (fi >= 0) {
   330→                        int start = fi + 10;
   331→                        int end = cd.indexOf("\"", start);
   332→                        if (end > start) originalName = cd.substring(start, end);
   333→                    }
   334→                }
   335→            }
   336→
   337→            String mimeType = getMimeType(originalName);
   338→            long fileSize = tempFile.length();
   339→
   340→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
   341→                // Android 10+ (Scoped Storage): use MediaStore ContentResolver
   342→                // This writes to the PUBLIC Downloads folder and is visible to all apps
   343→                ContentValues values = new ContentValues();
   344→                values.put(MediaStore.Downloads.DISPLAY_NAME, originalName);
   345→                values.put(MediaStore.Downloads.MIME_TYPE, mimeType);
   346→                values.put(MediaStore.Downloads.SIZE, fileSize);
   347→                values.put(MediaStore.Downloads.IS_PENDING, 1);
   348→                values.put(MediaStore.Downloads.RELATIVE_PATH, Environment.DIRECTORY_DOWNLOADS + "/SplannesFileShare");
   349→
   350→                Uri uri = context.getContentResolver().insert(MediaStore.Downloads.EXTERNAL_CONTENT_URI, values);
   351→                if (uri == null) {
   352→                    // Fallback: save to app external files
   353→                    Log.w(TAG, "MediaStore insert returned null, using app dir");
   354→                    saveToFallbackDir(tempFile, originalName, mimeType, fileSize);
   355→                } else {
   356→                    OutputStream os = context.getContentResolver().openOutputStream(uri);
   357→                    if (os != null) {
   358→                        copyStream(new FileInputStream(tempFile), os);
   359→                        os.close();
   360→                        // Clear IS_PENDING so the file is visible
   361→                        values.clear();
   362→                        values.put(MediaStore.Downloads.IS_PENDING, 0);
   363→                        context.getContentResolver().update(uri, values, null, null);
   364→                        Log.d(TAG, "Saved to Downloads via MediaStore: " + originalName);
   365→                    } else {
   366→                        Log.e(TAG, "ContentResolver openOutputStream returned null");
   367→                        saveToFallbackDir(tempFile, originalName, mimeType, fileSize);
   368→                    }
   369→                }
   370→            } else {
   371→                // Android 9 and below: direct file access
   372→                File downloadsDir = Environment.getExternalStoragePublicDirectory(Environment.DIRECTORY_DOWNLOADS);
   373→                File subDir = new File(downloadsDir, "SplannesFileShare");
   374→                if (!subDir.exists()) subDir.mkdirs();
   375→                File destFile = new File(subDir, originalName);
   376→                int suffix = 1;
   377→                while (destFile.exists()) {
   378→                    String base = originalName.contains(".") ? originalName.substring(0, originalName.lastIndexOf('.')) : originalName;
   379→                    String ext = originalName.contains(".") ? originalName.substring(originalName.lastIndexOf('.')) : "";
   380→                    destFile = new File(subDir, base + " (" + suffix + ")" + ext);
   381→                    suffix++;
   382→                }
   383→                copyFile(tempFile, destFile);
   384→                Log.d(TAG, "Saved to Downloads (direct): " + destFile.getAbsolutePath());
   385→            }
   386→
   387→            tempFile.delete();
   388→
   389→            // Track for the "Received on Phone" web list
   390→            receivedFiles.add(0, new ReceivedFile(originalName, fileSize));
   391→            while (receivedFiles.size() > 100) receivedFiles.remove(receivedFiles.size() - 1);
   392→
   393→            logTransfer("Uploaded: " + originalName + " (" + formatSize(fileSize) + ")");
   394→
   395→            return cors(newFixedLengthResponse(Response.Status.OK, "application/json",
   396→                    String.format("{\"success\":true,\"filename\":\"%s\",\"size\":%d}", esc(originalName), fileSize)));
   397→        } catch (Exception e) {
   398→            Log.e(TAG, "Upload error", e);
   399→            return cors(newFixedLengthResponse(Response.Status.INTERNAL_ERROR, "application/json",
   400→                    "{\"error\":\"" + esc(e.getMessage() != null ? e.getMessage() : "unknown") + "\"}"));
   401→        }
   402→    }

   404→    /** Fallback for when MediaStore fails — saves to app external files dir */
   405→    private void saveToFallbackDir(File tempFile, String originalName, String mimeType, long fileSize) throws IOException {
   406→        File dir = new File(context.getExternalFilesDir(null), "received");
   407→        if (!dir.exists()) dir.mkdirs();
   408→        File dest = new File(dir, originalName);
   409→        int s = 1;
   410→        while (dest.exists()) {
   411→            String base = originalName.contains(".") ? originalName.substring(0, originalName.lastIndexOf('.')) : originalName;
   412→            String ext = originalName.contains(".") ? originalName.substring(originalName.lastIndexOf('.')) : "";
   413→            dest = new File(dir, base + " (" + s + ")" + ext);
   414→            s++;
   415→        }
   416→        copyFile(tempFile, dest);
   417→        Log.d(TAG, "Fallback save: " + dest.getAbsolutePath());
   418→    }

   420→    private void copyStream(java.io.InputStream in, OutputStream out) throws IOException {
   421→        byte[] buf = new byte[8192];
   422→        int len;
   423→        while ((len = in.read(buf)) > 0) out.write(buf, 0, len);
   424→    }

   426→    /**
   427→     * Simple frame polling endpoint — returns the latest JPEG frame as a single HTTP response.
   428→     * The browser polls this every ~150ms by updating the img src with a cache-buster.
   429→     * Much more reliable than MJPEG streaming with NanoHTTPD 2.3.1.
   430→     */
   431→    private Response serveScreenFrame() {
   432→        if (!screenSharingActive) {
   433→            return cors(newFixedLengthResponse(Response.Status.OK, "application/json", "{\"active\":false}"));
   434→        }
   435→        byte[] frame;
   436→        synchronized (frameLock) { frame = latestScreenFrame; }
   437→        if (frame == null || frame.length == 0) {
   438→            // Return empty 200 with json so the JS knows to keep polling
   439→            return cors(newFixedLengthResponse(Response.Status.OK, "application/json", "{\"waiting\":true}"));
   440→        }
   441→        Response r = newChunkedResponse(Response.Status.OK, "image/jpeg", new ByteArrayInputStream(frame));
   442→        r.addHeader("Cache-Control", "no-cache, no-store, must-revalidate");
   443→        r.addHeader("Content-Length", String.valueOf(frame.length));
   444→        addCorsHeaders(r);
   445→        return r;
   446→    }

   503→    private long getFolderSize(File folder) {
   504→        long size = 0;
   505→        File[] files = folder.listFiles();
   506→        if (files != null) {
   507→            for (File f : files) {
   508→                size += f.isDirectory() ? getFolderSize(f) : f.length();
   509→            }
   510→        }
   511→        return size;

   514→    private void zipFolder(File folder, File zipFile) throws IOException {
   515→        ZipOutputStream zos = new ZipOutputStream(new FileOutputStream(zipFile));
   516→        zipRecursive(folder, folder.getName(), zos);
   517→        zos.close();
   518→    }
   519→
   520→    private void zipRecursive(File file, String entry, ZipOutputStream zos) throws IOException {
   521→        if (file.isDirectory()) {
   522→            File[] children = file.listFiles();
   523→            if (children != null) {
   524→                for (File child : children) zipRecursive(child, entry + "/" + child.getName(), zos);
   525→            }
   526→        } else {
   527→            zos.putNextEntry(new ZipEntry(entry));
   528→            FileInputStream fis = new FileInputStream(file);
   529→            byte[] buf = new byte[8192];
   530→            int len;
   531→            while ((len = fis.read(buf)) > 0) zos.write(buf, 0, len);
   532→            fis.close();
   533→            zos.closeEntry();
   534→        }
   535→    }

   537→    private void copyFile(File src, File dst) throws IOException {
   538→        FileInputStream fis = new FileInputStream(src);
   539→        FileOutputStream fos = new FileOutputStream(dst);
   540→        byte[] buf = new byte[8192];
   541→        int len;
   542→        while ((len = fis.read(buf)) > 0) fos.write(buf, 0, len);
   543→        fos.close();
   544→        fis.close();
   545→    }

   547→    private String getMimeType(String name) {
   548→        String n = name.toLowerCase(Locale.US);
   549→        if (n.endsWith(".html")) return "text/html";
   550→        if (n.endsWith(".css")) return "text/css";
   551→        if (n.endsWith(".js")) return "application/javascript";
   552→        if (n.endsWith(".json")) return "application/json";
   553→        if (n.endsWith(".png")) return "image/png";
   554→        if (n.endsWith(".jpg") || n.endsWith(".jpeg")) return "image/jpeg";
   555→        if (n.endsWith(".gif")) return "image/gif";
   556→        if (n.endsWith(".mp4")) return "video/mp4";
   557→        if (n.endsWith(".mp3")) return "audio/mpeg";
   558→        if (n.endsWith(".pdf")) return "application/pdf";
   559→        if (n.endsWith(".zip")) return "application/zip";
   560→        if (n.endsWith(".apk")) return "application/vnd.android.package-archive";
   561→        if (n.endsWith(".txt")) return "text/plain";
   562→        return "application/octet-stream";
   563→    }

   565→    private String formatSize(long bytes) {
   566→        if (bytes < 1024) return bytes + " B";
   567→        if (bytes < 1048576) return String.format(Locale.US, "%.1f KB", bytes / 1024.0);
   568→        if (bytes < 1073741824) return String.format(Locale.US, "%.1f MB", bytes / (1048576.0));
   569→        return String.format(Locale.US, "%.1f GB", bytes / (1073741824.0));
   570→    }
