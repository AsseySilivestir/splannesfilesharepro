   173→    private void checkAndRequestPermissions() {
   174→        List<String> permissionsNeeded = new ArrayList<>();
   175→
   176→        if (ContextCompat.checkSelfPermission(this, Manifest.permission.INTERNET) != PackageManager.PERMISSION_GRANTED) {
   177→            permissionsNeeded.add(Manifest.permission.INTERNET);
   178→        }
   179→        if (ContextCompat.checkSelfPermission(this, Manifest.permission.ACCESS_NETWORK_STATE) != PackageManager.PERMISSION_GRANTED) {
   180→            permissionsNeeded.add(Manifest.permission.ACCESS_NETWORK_STATE);
   181→        }
   182→        if (ContextCompat.checkSelfPermission(this, Manifest.permission.ACCESS_WIFI_STATE) != PackageManager.PERMISSION_GRANTED) {
   183→            permissionsNeeded.add(Manifest.permission.ACCESS_WIFI_STATE);
   184→        }
   185→        if (ContextCompat.checkSelfPermission(this, Manifest.permission.CHANGE_WIFI_STATE) != PackageManager.PERMISSION_GRANTED) {
   186→            permissionsNeeded.add(Manifest.permission.CHANGE_WIFI_STATE);
   187→        }
   188→        if (ContextCompat.checkSelfPermission(this, Manifest.permission.FOREGROUND_SERVICE) != PackageManager.PERMISSION_GRANTED) {
   189→            permissionsNeeded.add(Manifest.permission.FOREGROUND_SERVICE);
   190→        }
   191→        if (ContextCompat.checkSelfPermission(this, Manifest.permission.POST_NOTIFICATIONS) != PackageManager.PERMISSION_GRANTED) {
   192→            permissionsNeeded.add(Manifest.permission.POST_NOTIFICATIONS);
   193→        }
   194→
   195→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
   196→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_IMAGES) != PackageManager.PERMISSION_GRANTED) {
   197→                permissionsNeeded.add(Manifest.permission.READ_MEDIA_IMAGES);
   198→            }
   199→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_VIDEO) != PackageManager.PERMISSION_GRANTED) {
   200→                permissionsNeeded.add(Manifest.permission.READ_MEDIA_VIDEO);
   201→            }
   202→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_AUDIO) != PackageManager.PERMISSION_GRANTED) {
   203→                permissionsNeeded.add(Manifest.permission.READ_MEDIA_AUDIO);
   204→            }
   205→        } else {
   206→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_EXTERNAL_STORAGE) != PackageManager.PERMISSION_GRANTED) {
   207→                permissionsNeeded.add(Manifest.permission.READ_EXTERNAL_STORAGE);
   208→            }
   209→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.WRITE_EXTERNAL_STORAGE) != PackageManager.PERMISSION_GRANTED) {
   210→                permissionsNeeded.add(Manifest.permission.WRITE_EXTERNAL_STORAGE);
   211→            }
   212→        }
   213→
   214→        // Request MANAGE_EXTERNAL_STORAGE for folder access on Android 11+
   215→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R) {
   216→            if (!Environment.isExternalStorageManager()) {
   217→                try {
   218→                    Intent intent = new Intent(android.provider.Settings.ACTION_MANAGE_APP_ALL_FILES_ACCESS_PERMISSION,
   219→                        Uri.parse("package:" + getPackageName()));
   220→                    startActivity(intent);
   221→                } catch (Exception e) {
   222→                    Intent intent = new Intent(android.provider.Settings.ACTION_MANAGE_ALL_FILES_ACCESS_PERMISSION);
   223→                    startActivity(intent);
   224→                }
   225→            }
   226→        }
   227→
   228→        if (!permissionsNeeded.isEmpty()) {
   229→            ActivityCompat.requestPermissions(this,
   230→                permissionsNeeded.toArray(new String[0]),
   231→                PERMISSION_REQUEST_CODE);
   232→        }
   233→    }

   362→    private void handleFilesSelected(Intent data) {
   363→        List<Uri> uris = new ArrayList<>();
   364→        if (data.getClipData() != null) {
   365→            int count = data.getClipData().getItemCount();
   366→            for (int i = 0; i < count; i++) {
   367→                uris.add(data.getClipData().getItemAt(i).getUri());
   368→            }
   369→        } else if (data.getData() != null) {
   370→            uris.add(data.getData());
   371→        }
   372→
   373→        if (uris.isEmpty()) return;
   374→
   375→        // Copy URIs to local files for the server to serve
   376→        File shareDir = new File(getExternalFilesDir(null), "shared");
   377→        if (!shareDir.exists()) shareDir.mkdirs();
   378→
   379→        List<File> sharedFiles = new ArrayList<>();
   380→        for (Uri uri : uris) {
   381→            try {
   382→                String fileName = getFileName(uri);
   383→                File destFile = new File(shareDir, fileName);
   384→                // Avoid overwriting
   385→                int suffix = 1;
   386→                while (destFile.exists()) {
   387→                    String baseName = fileName.contains(".")
   388→                        ? fileName.substring(0, fileName.lastIndexOf('.'))
   389→                        : fileName;
   390→                    String ext = fileName.contains(".")
   391→                        ? fileName.substring(fileName.lastIndexOf('.'))
   392→                        : "";
   393→                    destFile = new File(shareDir, baseName + " (" + suffix + ")" + ext);
   394→                    suffix++;
   395→                }
   396→                copyUriToFile(uri, destFile);
   397→                sharedFiles.add(destFile);
   398→            } catch (Exception e) {
   399→                Log.e(TAG, "Error copying file", e);
   400→                Toast.makeText(this, "Error: " + e.getMessage(), Toast.LENGTH_SHORT).show();
   401→            }
   402→        }
   403→
   404→        if (fileShareService != null && fileShareService.getServer() != null) {
   405→            List<Uri> fileUris = new ArrayList<>();
   406→            for (File f : sharedFiles) {
   407→                fileUris.add(Uri.fromFile(f));
   408→            }
   409→            fileShareService.getServer().addSharedFiles(fileUris);
   410→        }
   411→
   412→        tvSelectedInfo.setVisibility(View.VISIBLE);
   413→        tvSelectedInfo.setText(getString(R.string.files_selected, sharedFiles.size()));
   414→        Toast.makeText(this, sharedFiles.size() + " file(s) shared successfully", Toast.LENGTH_SHORT).show();
   415→    }

   425→    private void handleFolderSelected(Uri uri) {

   427→        File shareDir = new File(getExternalFilesDir(null), "shared");

   429→
   430→        try {

   435→            copyFolderFromUri(uri, localFolder);

   462→    private void requestScreenCapture() {
   463→        if (!screenBound) {
   464→            startScreenShareService();
   465→            // Wait a moment for binding
   466→            new android.os.Handler().postDelayed(this::requestScreenCapture, 500);
   467→            return;
   468→        }
   469→
   470→        android.media.projection.MediaProjectionManager projectionManager =
   471→            (android.media.projection.MediaProjectionManager) getSystemService(Context.MEDIA_PROJECTION_SERVICE);
   472→        Intent intent = projectionManager.createScreenCaptureIntent();
   473→        screenCaptureLauncher.launch(intent);

   476→    private void startScreenSharing(int resultCode, Intent data) {
   477→        if (screenShareService == null) return;
   478→
   479→        screenShareService.startCapture(resultCode, data);
   480→        screenSharing = true;
   481→
   482→        btnScreenShare.setText(R.string.stop_sharing_screen);
   483→        btnScreenShare.setBackgroundColor(getColor(R.color.red));
   484→
   485→        // Connect screen capture to server
   486→        if (fileShareService != null && fileShareService.getServer() != null) {
   487→            fileShareService.getServer().setScreenSharingActive(true);
   488→            fileShareService.getServer().setScreenCaptureCallback(() -> {
   489→                // Return the latest frame
   490→                // The ScreenShareService handles frame capture via its callback
   491→                return screenShareService.isCapturing() ? new byte[]{(byte) 0xFF} : null;
   492→            });
   493→
   494→            // Set up the proper frame callback on the screen share service
   495→            screenShareService.setFrameCallback(jpegBytes -> {
   496→                // Store the latest frame for the server to pick up
   497→                return jpegBytes;
   498→            });
   499→        }
   500→
   501→        // We need to properly connect the screen service frames to the server
   502→        // Let's set up a direct frame reference
   503→        final byte[][] latestFrame = {null};
   504→        screenShareService.setFrameCallback(jpegBytes -> {
   505→            latestFrame[0] = jpegBytes;
   506→            return jpegBytes;
   507→        });
   508→
   509→        if (fileShareService != null && fileShareService.getServer() != null) {
   510→            fileShareService.getServer().setScreenCaptureCallback(() -> latestFrame[0]);
   511→        }
   512→
   513→        Toast.makeText(this, "Screen sharing started", Toast.LENGTH_SHORT).show();
   514→    }

   516→    private void stopScreenSharing() {

   520→        if (fileShareService != null && fileShareService.getServer() != null) {
   521→            fileShareService.getServer().setScreenSharingActive(false);
   522→            fileShareService.getServer().setScreenCaptureCallback(null);
   523→        }
   524→
   525→        screenSharing = false;
   526→        btnScreenShare.setText(R.string.share_screen);
   527→        btnScreenShare.setBackgroundColor(getColor(R.color.accent));
   528→
   529→        Toast.makeText(this, "Screen sharing stopped", Toast.LENGTH_SHORT).show();
   530→    }

   534→    private String getLocalIpAddress() {
   535→        try {
   536→            for (NetworkInterface intf : Collections.list(NetworkInterface.getNetworkInterfaces())) {
   537→                for (InetAddress addr : Collections.list(intf.getInetAddresses())) {
   538→                    if (!addr.isLoopbackAddress() && addr.getHostAddress().indexOf(':') < 0) {
   539→                        String ip = addr.getHostAddress();
   540→                        if (ip.startsWith("192.168.") || ip.startsWith("10.") || ip.startsWith("172.")) {
   541→                            return ip;
   542→                        }
   543→                    }
   544→                }
   545→            }
   546→            // Fallback: try WiFi manager
   547→            WifiManager wifiManager = (WifiManager) getApplicationContext().getSystemService(Context.WIFI_SERVICE);
   548→            if (wifiManager != null) {
   549→                WifiInfo wifiInfo = wifiManager.getConnectionInfo();
   550→                int ipInt = wifiInfo.getIpAddress();
   551→                if (ipInt != 0) {
   552→                    return String.format("%d.%d.%d.%d",
   553→                        (ipInt & 0xFF), (ipInt >> 8 & 0xFF),
   554→                        (ipInt >> 16 & 0xFF), (ipInt >> 24 & 0xFF));
   555→                }
   556→            }
   557→        } catch (Exception e) {
   558→            Log.e(TAG, "Error getting IP", e);
   559→        }
   560→        return "0.0.0.0";
   561→    }

   563→    private String getFileName(Uri uri) {
   564→        String fileName = "file";
   565→        try {
   566→            String displayName = null;
   567→
   568→            if (uri.getScheme() != null && uri.getScheme().equals("content")) {
   569→                try (android.database.Cursor cursor = getContentResolver().query(uri, null, null, null, null)) {
   570→                    if (cursor != null && cursor.moveToFirst()) {
   571→                        int idx = cursor.getColumnIndex(OpenableColumns.DISPLAY_NAME);
   572→                        if (idx >= 0) {
   573→                            displayName = cursor.getString(idx);
   574→                        }
   575→                    }
   576→                }
   577→            }
   578→
   579→            if (displayName == null) {
   580→                String path = uri.getPath();
   581→                if (path != null) {
   582→                    int lastSlash = path.lastIndexOf('/');
   583→                    if (lastSlash >= 0) {
   584→                        fileName = path.substring(lastSlash + 1);
   585→                    }
   586→                }
   587→            } else {
   588→                fileName = displayName;
   589→            }
   590→        } catch (Exception e) {
   591→            Log.e(TAG, "Error getting file name", e);
   592→        }
   593→
   594→        if (TextUtils.isEmpty(fileName)) fileName = "file";
   595→        return fileName;
   596→    }

   598→    private String getFolderName(Uri uri) {
   599→        // Try to get a nice name from the tree URI
   600→        String lastPath = uri.getLastPathSegment();
   601→        if (lastPath != null) {
   602→            int colonIdx = lastPath.lastIndexOf(':');
   603→            if (colonIdx >= 0) {
   604→                lastPath = lastPath.substring(colonIdx + 1);
   605→            }
   606→            if (!TextUtils.isEmpty(lastPath)) {
   607→                return lastPath.replace("/", "_");
   608→            }
   609→        }
   610→        return "shared_folder_" + System.currentTimeMillis();
   611→    }

   613→    private void copyUriToFile(Uri uri, File destFile) throws Exception {
   614→        try (InputStream in = getContentResolver().openInputStream(uri);
   615→             OutputStream out = new FileOutputStream(destFile)) {
   616→            if (in == null) throw new Exception("Cannot open source file");
   617→            byte[] buffer = new byte[8192];
   618→            int len;
   619→            while ((len = in.read(buffer)) > 0) {
   620→                out.write(buffer, 0, len);
   621→            }
   622→        }
   623→    }

   625→    private void copyFolderFromUri(Uri treeUri, File destFolder) throws Exception {
   626→        // List and copy all children of the tree URI
   627→        String treeDocId = DocumentsContract.getTreeDocumentId(treeUri);
   628→        Uri childrenUri = DocumentsContract.buildChildDocumentsUriUsingTree(treeUri, treeDocId);
   629→
   630→        try (android.database.Cursor cursor = getContentResolver().query(
   631→                childrenUri,
   632→                new String[]{DocumentsContract.Document.COLUMN_DOCUMENT_ID,
   633→                    DocumentsContract.Document.COLUMN_DISPLAY_NAME,
   634→                    DocumentsContract.Document.COLUMN_MIME_TYPE},
   635→                null, null, null)) {
   636→
   637→            if (cursor != null) {
   638→                while (cursor.moveToNext()) {
   639→                    String docId = cursor.getString(0);
   640→                    String name = cursor.getString(1);
   641→                    String mimeType = cursor.getString(2);
   642→
   643→                    Uri docUri = DocumentsContract.buildDocumentUriUsingTree(treeUri, docId);
   644→                    File outFile = new File(destFolder, name);
   645→
   646→                    if (DocumentsContract.Document.MIME_TYPE_DIR.equals(mimeType)) {
   647→                        outFile.mkdirs();
   648→                        copyFolderFromUri(docUri, outFile);
   649→                    } else {
   650→                        copyUriToFile(docUri, outFile);
   651→                    }
   652→                }
   653→            }
   654→        }
   655→    }
