   163→    private void requestAllPermissions() {
   164→        List<String> needed = new ArrayList<>();
   165→
   166→        if (ContextCompat.checkSelfPermission(this, Manifest.permission.POST_NOTIFICATIONS) != PackageManager.PERMISSION_GRANTED) {
   167→            needed.add(Manifest.permission.POST_NOTIFICATIONS);
   168→        }
   169→
   170→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
   171→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_IMAGES) != PackageManager.PERMISSION_GRANTED)
   172→                needed.add(Manifest.permission.READ_MEDIA_IMAGES);
   173→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_VIDEO) != PackageManager.PERMISSION_GRANTED)
   174→                needed.add(Manifest.permission.READ_MEDIA_VIDEO);
   175→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_AUDIO) != PackageManager.PERMISSION_GRANTED)
   176→                needed.add(Manifest.permission.READ_MEDIA_AUDIO);
   177→        } else {
   178→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_EXTERNAL_STORAGE) != PackageManager.PERMISSION_GRANTED)
   179→                needed.add(Manifest.permission.READ_EXTERNAL_STORAGE);
   180→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.WRITE_EXTERNAL_STORAGE) != PackageManager.PERMISSION_GRANTED)
   181→                needed.add(Manifest.permission.WRITE_EXTERNAL_STORAGE);
   182→        }
   183→
   184→        if (!needed.isEmpty()) {
   185→            ActivityCompat.requestPermissions(this, needed.toArray(new String[0]), PERM_REQ_GENERAL);
   186→        }
   187→
   188→        // MANAGE_EXTERNAL_STORAGE for full folder access (Android 11+)
   189→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R && !Environment.isExternalStorageManager()) {
   190→            try {
   191→                startActivity(new Intent(Settings.ACTION_MANAGE_APP_ALL_FILES_ACCESS_PERMISSION,
   192→                    Uri.parse("package:" + getPackageName())));
   193→            } catch (Exception e) {
   194→                startActivity(new Intent(Settings.ACTION_MANAGE_ALL_FILES_ACCESS_PERMISSION));
   195→            }
   196→        }
   197→    }

   199→    @Override
   200→    public void onRequestPermissionsResult(int requestCode, @NonNull String[] permissions, @NonNull int[] grantResults) {
   201→        super.onRequestPermissionsResult(requestCode, permissions, grantResults);
   202→        if (requestCode == PERM_REQ_SCREEN) {
   203→            // All screen-share permissions granted → now show the system capture dialog
   204→            if (allGranted(grantResults)) {
   205→                showScreenCaptureDialog();
   206→            } else {
   207→                Toast.makeText(this, "Screen share permission denied", Toast.LENGTH_SHORT).show();
   208→            }
   209→        }
   210→    }

   212→    private boolean allGranted(int[] results) {
   213→        for (int r : results) if (r != PackageManager.PERMISSION_GRANTED) return false;
   214→        return true;
   215→    }

   235→        if (!serverBound || fileShareService == null) {

   312→    private void handleFilesSelected(Intent data) {
   313→        List<Uri> uris = new ArrayList<>();
   314→        if (data.getClipData() != null) {
   315→            int count = data.getClipData().getItemCount();
   316→            for (int i = 0; i < count; i++) uris.add(data.getClipData().getItemAt(i).getUri());
   317→        } else if (data.getData() != null) {
   318→            uris.add(data.getData());
   319→        }
   320→        if (uris.isEmpty()) return;
   321→
   322→        Toast.makeText(this, "Copying " + uris.size() + " file(s)...", Toast.LENGTH_SHORT).show();
   323→
   324→        // Copy files on background thread to prevent ANR
   325→        ioExecutor.execute(() -> {
   326→            File shareDir = new File(getExternalFilesDir(null), "shared");
   327→            if (!shareDir.exists()) shareDir.mkdirs();
   328→
   329→            List<String> copiedPaths = new ArrayList<>();
   330→            for (Uri uri : uris) {
   331→                try {
   332→                    String fileName = getFileName(uri);
   333→                    if (fileName == null || fileName.isEmpty()) fileName = "file_" + System.currentTimeMillis();
   334→
   335→                    File destFile = new File(shareDir, fileName);
   336→                    int suffix = 1;
   337→                    while (destFile.exists()) {
   338→                        String base = fileName.contains(".") ? fileName.substring(0, fileName.lastIndexOf('.')) : fileName;
   339→                        String ext = fileName.contains(".") ? fileName.substring(fileName.lastIndexOf('.')) : "";
   340→                        destFile = new File(shareDir, base + " (" + suffix + ")" + ext);
   341→                        suffix++;
   342→                    }
   343→
   344→                    copyUriToFile(uri, destFile);
   345→                    copiedPaths.add(destFile.getAbsolutePath());
   346→                } catch (Exception e) {
   347→                    Log.e(TAG, "Error copying file: " + uri, e);
   348→                    runOnUiThread(() -> Toast.makeText(this, "Error: " + e.getMessage(), Toast.LENGTH_SHORT).show());
   349→                }
   350→            }
   351→
   352→            // Add to server on the IO thread (server uses thread-safe collections)
   353→            if (!copiedPaths.isEmpty() && fileShareService != null && fileShareService.getServer() != null) {
   354→                fileShareService.getServer().addSharedFilePaths(copiedPaths);
   355→            }
   356→
   357→            int count = copiedPaths.size();
   358→            runOnUiThread(() -> {
   359→                tvSelectedInfo.setVisibility(View.VISIBLE);
   360→                tvSelectedInfo.setText(getString(R.string.files_selected, count));
   361→                if (count > 0) {
   362→                    Toast.makeText(this, count + " file(s) shared successfully!", Toast.LENGTH_SHORT).show();
   363→                } else {
   364→                    Toast.makeText(this, "No files were shared", Toast.LENGTH_SHORT).show();
   365→                }
   366→            });
   367→        });
   368→    }

   380→    private void handleFolderSelected(Uri uri) {
   381→        Toast.makeText(this, "Copying folder...", Toast.LENGTH_SHORT).show();
   382→
   383→        ioExecutor.execute(() -> {
   384→            try {
   385→                File shareDir = new File(getExternalFilesDir(null), "shared");
   386→                if (!shareDir.exists()) shareDir.mkdirs();
   387→
   388→                String folderName = getFolderName(uri);
   389→                File localFolder = new File(shareDir, folderName);
   390→                if (!localFolder.exists()) localFolder.mkdirs();
   391→
   392→                copyFolderFromUri(uri, localFolder);
   393→
   394→                if (fileShareService != null && fileShareService.getServer() != null) {
   395→                    fileShareService.getServer().addSharedFolderPath(localFolder.getAbsolutePath());
   396→                }
   397→
   398→                runOnUiThread(() -> {
   399→                    tvSelectedInfo.setVisibility(View.VISIBLE);
   400→                    tvSelectedInfo.setText("Folder shared: " + folderName);
   401→                    Toast.makeText(this, "Folder shared: " + folderName, Toast.LENGTH_SHORT).show();
   402→                });
   403→            } catch (Exception e) {
   404→                Log.e(TAG, "Error sharing folder", e);
   405→                runOnUiThread(() -> Toast.makeText(this, "Error: " + e.getMessage(), Toast.LENGTH_SHORT).show());
   406→            }
   407→        });
   408→    }

   412→    /**
   413→     * Correct Android 14 flow:
   414→     * 1. Request FOREGROUND_SERVICE_MEDIA_PROJECTION permission
   415→     * 2. Show system MediaProjection consent dialog
   416→     * 3. On consent result → start foreground service with result in Intent extras
   417→     */
   418→    private void toggleScreenShare() {
   419→        if (screenSharing) {
   420→            stopScreenSharing();
   421→            return;
   422→        }
   423→        if (!serverRunning) {
   424→            Toast.makeText(this, "Start the server first", Toast.LENGTH_SHORT).show();
   425→            return;
   426→        }
   427→
   428→        // Step 1: Check/request FOREGROUND_SERVICE_MEDIA_PROJECTION
   429→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.UPSIDE_DOWN_CAKE) {
   430→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.FOREGROUND_SERVICE_MEDIA_PROJECTION) != PackageManager.PERMISSION_GRANTED) {
   431→                ActivityCompat.requestPermissions(this,
   432→                    new String[]{Manifest.permission.FOREGROUND_SERVICE_MEDIA_PROJECTION},
   433→                    PERM_REQ_SCREEN);
   434→                return; // result comes back in onRequestPermissionsResult
   435→            }
   436→        }
   437→
   438→        showScreenCaptureDialog();
   439→    }

   441→    /** Step 2: Show the system screen capture consent dialog */
   442→    private void showScreenCaptureDialog() {
   443→        android.media.projection.MediaProjectionManager pm =
   444→            (android.media.projection.MediaProjectionManager) getSystemService(Context.MEDIA_PROJECTION_SERVICE);
   445→        screenCaptureLauncher.launch(pm.createScreenCaptureIntent());

   448→    /** Step 3: User granted consent → start foreground service WITH the result */
   449→    private void launchScreenShareService(int resultCode, Intent data) {
   450→        try {
   451→            Intent serviceIntent = new Intent(this, ScreenShareService.class);
   452→            serviceIntent.putExtra(ScreenShareService.EXTRA_RESULT_CODE, resultCode);
   453→            serviceIntent.putExtra("data", data);
   454→
   455→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
   456→                startForegroundService(serviceIntent);
   457→            } else {
   458→                startService(serviceIntent);
   459→            }
   460→
   461→            // Connect the screen service to the file server for frame delivery
   462→            if (fileShareService != null && fileShareService.getServer() != null) {
   463→                fileShareService.getServer().setScreenSharingActive(true);
   464→                // Bind briefly to wire up the server reference
   465→                bindService(new Intent(this, ScreenShareService.class), new ServiceConnection() {
   466→                    @Override
   467→                    public void onServiceConnected(ComponentName name, IBinder service) {
   468→                        ScreenShareService screenSvc = ((ScreenShareService.LocalBinder) service).getService();
   469→                        screenSvc.setFileShareServer(fileShareService.getServer());
   470→                        try { unbindService(this); } catch (Exception ignored) {}
   471→                    }
   472→                    @Override
   473→                    public void onServiceDisconnected(ComponentName name) {}
   474→                }, Context.BIND_AUTO_CREATE);
   475→            }
   476→
   477→            screenSharing = true;
   478→            btnScreenShare.setText(R.string.stop_sharing_screen);
   479→            btnScreenShare.setBackgroundColor(getColor(R.color.red));
   480→            Toast.makeText(this, "Screen sharing started", Toast.LENGTH_SHORT).show();
   481→
   482→        } catch (Exception e) {
   483→            Log.e(TAG, "Failed to start screen share", e);
   484→            Toast.makeText(this, "Failed: " + e.getMessage(), Toast.LENGTH_SHORT).show();
   485→        }
   486→    }

   488→    private void stopScreenSharing() {
   489→        try {
   490→            stopService(new Intent(this, ScreenShareService.class));
   491→        } catch (Exception ignored) {}
   492→
   493→        if (fileShareService != null && fileShareService.getServer() != null) {
   494→            fileShareService.getServer().setScreenSharingActive(false);
   495→            fileShareService.getServer().setScreenCaptureCallback(null);
   496→        }
   497→
   498→        screenSharing = false;
   499→        btnScreenShare.setText(R.string.share_screen);
   500→        btnScreenShare.setBackgroundColor(getColor(R.color.accent));
   501→        Toast.makeText(this, "Screen sharing stopped", Toast.LENGTH_SHORT).show();
   502→    }

   506→    private String getLocalIpAddress() {
   507→        try {
   508→            for (NetworkInterface intf : Collections.list(NetworkInterface.getNetworkInterfaces())) {
   509→                for (InetAddress addr : Collections.list(intf.getInetAddresses())) {
   510→                    if (!addr.isLoopbackAddress() && addr.getHostAddress().indexOf(':') < 0) {
   511→                        String ip = addr.getHostAddress();
   512→                        if (ip.startsWith("192.168.") || ip.startsWith("10.") || ip.startsWith("172.")) {
   513→                            return ip;
   514→                        }
   515→                    }
   516→                }
   517→            }
   518→            WifiManager wm = (WifiManager) getApplicationContext().getSystemService(Context.WIFI_SERVICE);
   519→            if (wm != null) {
   520→                WifiInfo wifiInfo = wm.getConnectionInfo();
   521→                int ipInt = wifiInfo.getIpAddress();
   522→                if (ipInt != 0) {
   523→                    return String.format("%d.%d.%d.%d",
   524→                        (ipInt & 0xFF), (ipInt >> 8 & 0xFF),
   525→                        (ipInt >> 16 & 0xFF), (ipInt >> 24 & 0xFF));
   526→                }
   527→            }
   528→        } catch (Exception e) {
   529→            Log.e(TAG, "Error getting IP", e);
   530→        }
   531→        return "0.0.0.0";
   532→    }

   534→    private String getFileName(Uri uri) {
   535→        try {
   536→            if (uri != null && "content".equals(uri.getScheme())) {
   537→                try (android.database.Cursor cursor = getContentResolver().query(uri,
   538→                        new String[]{OpenableColumns.DISPLAY_NAME}, null, null, null)) {
   539→                    if (cursor != null && cursor.moveToFirst()) {
   540→                        int idx = cursor.getColumnIndex(OpenableColumns.DISPLAY_NAME);
   541→                        if (idx >= 0) {
   542→                            String name = cursor.getString(idx);
   543→                            if (!TextUtils.isEmpty(name)) return name;
   544→                        }
   545→                    }
   546→                }
   547→            }
   548→            if (uri != null && uri.getPath() != null) {
   549→                String path = uri.getPath();
   550→                int slash = path.lastIndexOf('/');
   551→                if (slash >= 0) return path.substring(slash + 1);
   552→            }
   553→        } catch (Exception e) {
   554→            Log.e(TAG, "Error getting file name", e);
   555→        }
   556→        return "file_" + System.currentTimeMillis();
   557→    }
   558→
   559→    private String getFolderName(Uri uri) {
   560→        try {
   561→            String lastPath = uri.getLastPathSegment();
   562→            if (lastPath != null) {
   563→                int colonIdx = lastPath.lastIndexOf(':');
   564→                if (colonIdx >= 0) lastPath = lastPath.substring(colonIdx + 1);
   565→                if (!TextUtils.isEmpty(lastPath)) return lastPath.replace("/", "_");
   566→            }
   567→        } catch (Exception e) {
   568→            Log.e(TAG, "Error getting folder name", e);
   569→        }
   570→        return "shared_folder_" + System.currentTimeMillis();
   571→    }
   572→
   573→    private void copyUriToFile(Uri uri, File destFile) throws Exception {
   574→        InputStream in = getContentResolver().openInputStream(uri);
   575→        if (in == null) throw new Exception("Cannot open file");
   576→        try (OutputStream out = new FileOutputStream(destFile)) {
   577→            byte[] buf = new byte[8192];
   578→            int len;
   579→            while ((len = in.read(buf)) > 0) out.write(buf, 0, len);
   580→        } finally {
   581→            in.close();
   582→        }
   583→    }
   584→
   585→    private void copyFolderFromUri(Uri treeUri, File destFolder) throws Exception {
   586→        String treeDocId = DocumentsContract.getTreeDocumentId(treeUri);
   587→        Uri childrenUri = DocumentsContract.buildChildDocumentsUriUsingTree(treeUri, treeDocId);
   588→
   589→        try (android.database.Cursor cursor = getContentResolver().query(childrenUri,
   590→                new String[]{
   591→                    DocumentsContract.Document.COLUMN_DOCUMENT_ID,
   592→                    DocumentsContract.Document.COLUMN_DISPLAY_NAME,
   593→                    DocumentsContract.Document.COLUMN_MIME_TYPE
   594→                }, null, null, null)) {
   595→
   596→            if (cursor != null) {
   597→                while (cursor.moveToNext()) {
   598→                    String docId = cursor.getString(0);
   599→                    String name = cursor.getString(1);
   600→                    String mimeType = cursor.getString(2);
   601→                    Uri docUri = DocumentsContract.buildDocumentUriUsingTree(treeUri, docId);
   602→                    File outFile = new File(destFolder, name);
   603→
   604→                    if (DocumentsContract.Document.MIME_TYPE_DIR.equals(mimeType)) {
   605→                        outFile.mkdirs();
   606→                        copyFolderFromUri(docUri, outFile);
   607→                    } else {
   608→                        copyUriToFile(docUri, outFile);
   609→                    }
   610→                }
   611→            }
   612→        }
   613→    }
