   163→   163→    private void requestAllPermissions() {
   164→   164→        List<String> needed = new ArrayList<>();
   165→   165→
   166→   166→        if (ContextCompat.checkSelfPermission(this, Manifest.permission.POST_NOTIFICATIONS) != PackageManager.PERMISSION_GRANTED) {
   167→   167→            needed.add(Manifest.permission.POST_NOTIFICATIONS);
   168→   168→        }
   169→   169→
   170→   170→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
   171→   171→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_IMAGES) != PackageManager.PERMISSION_GRANTED)
   172→   172→                needed.add(Manifest.permission.READ_MEDIA_IMAGES);
   173→   173→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_VIDEO) != PackageManager.PERMISSION_GRANTED)
   174→   174→                needed.add(Manifest.permission.READ_MEDIA_VIDEO);
   175→   175→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_AUDIO) != PackageManager.PERMISSION_GRANTED)
   176→   176→                needed.add(Manifest.permission.READ_MEDIA_AUDIO);
   177→   177→        } else {
   178→   178→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_EXTERNAL_STORAGE) != PackageManager.PERMISSION_GRANTED)
   179→   179→                needed.add(Manifest.permission.READ_EXTERNAL_STORAGE);
   180→   180→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.WRITE_EXTERNAL_STORAGE) != PackageManager.PERMISSION_GRANTED)
   181→   181→                needed.add(Manifest.permission.WRITE_EXTERNAL_STORAGE);
   182→   182→        }
   183→   183→
   184→   184→        if (!needed.isEmpty()) {
   185→   185→            ActivityCompat.requestPermissions(this, needed.toArray(new String[0]), PERM_REQ_GENERAL);
   186→   186→        }
   187→   187→
   188→   188→        // MANAGE_EXTERNAL_STORAGE for full folder access (Android 11+)
   189→   189→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R && !Environment.isExternalStorageManager()) {
   190→   190→            try {
   191→   191→                startActivity(new Intent(Settings.ACTION_MANAGE_APP_ALL_FILES_ACCESS_PERMISSION,
   192→   192→                    Uri.parse("package:" + getPackageName())));
   193→   193→            } catch (Exception e) {
   194→   194→                startActivity(new Intent(Settings.ACTION_MANAGE_ALL_FILES_ACCESS_PERMISSION));
   195→   195→            }
   196→   196→        }
   197→   197→    }

   312→   312→    private void handleFilesSelected(Intent data) {
   313→   313→        List<Uri> uris = new ArrayList<>();
   314→   314→        if (data.getClipData() != null) {
   315→   315→            int count = data.getClipData().getItemCount();
   316→   316→            for (int i = 0; i < count; i++) uris.add(data.getClipData().getItemAt(i).getUri());
   317→   317→        } else if (data.getData() != null) {
   318→   318→            uris.add(data.getData());
   319→   319→        }
   320→   320→        if (uris.isEmpty()) return;
   321→   321→
   322→   322→        Toast.makeText(this, "Copying " + uris.size() + " file(s)...", Toast.LENGTH_SHORT).show();
   323→   323→
   324→   324→        // Copy files on background thread to prevent ANR
   325→   325→        ioExecutor.execute(() -> {
   326→   326→            File shareDir = new File(getExternalFilesDir(null), "shared");
   327→   327→            if (!shareDir.exists()) shareDir.mkdirs();
   328→   328→
   329→   329→            List<String> copiedPaths = new ArrayList<>();
   330→   330→            for (Uri uri : uris) {
   331→   331→                try {
   332→   332→                    String fileName = getFileName(uri);
   333→   333→                    if (fileName == null || fileName.isEmpty()) fileName = "file_" + System.currentTimeMillis();
   334→   334→
   335→   335→                    File destFile = new File(shareDir, fileName);
   336→   336→                    int suffix = 1;
   337→   337→                    while (destFile.exists()) {
   338→   338→                        String base = fileName.contains(".") ? fileName.substring(0, fileName.lastIndexOf('.')) : fileName;
   339→   339→                        String ext = fileName.contains(".") ? fileName.substring(fileName.lastIndexOf('.')) : "";
   340→   340→                        destFile = new File(shareDir, base + " (" + suffix + ")" + ext);
   341→   341→                        suffix++;
   342→   342→                    }
   343→   343→
   344→   344→                    copyUriToFile(uri, destFile);
   345→   345→                    copiedPaths.add(destFile.getAbsolutePath());
   346→   346→                } catch (Exception e) {
   347→   347→                    Log.e(TAG, "Error copying file: " + uri, e);
   348→   348→                    runOnUiThread(() -> Toast.makeText(this, "Error: " + e.getMessage(), Toast.LENGTH_SHORT).show());
   349→   349→                }
   350→   350→            }
   351→   351→
   352→   352→            // Add to server on the IO thread (server uses thread-safe collections)
   353→   353→            if (!copiedPaths.isEmpty() && fileShareService != null && fileShareService.getServer() != null) {
   354→   354→                fileShareService.getServer().addSharedFilePaths(copiedPaths);
   355→   355→            }
   356→   356→
   357→   357→            int count = copiedPaths.size();
   358→   358→            runOnUiThread(() -> {
   359→   359→                tvSelectedInfo.setVisibility(View.VISIBLE);
   360→   360→                tvSelectedInfo.setText(getString(R.string.files_selected, count));
   361→   361→                if (count > 0) {
   362→   362→                    Toast.makeText(this, count + " file(s) shared successfully!", Toast.LENGTH_SHORT).show();
   363→   363→                } else {
   364→   364→                    Toast.makeText(this, "No files were shared", Toast.LENGTH_SHORT).show();
   365→   365→                }
   366→   366→            });
   367→   367→        });
   368→   368→    }

   380→   380→    private void handleFolderSelected(Uri uri) {

   386→   386→                if (!shareDir.exists()) shareDir.mkdirs();
   387→   387→

   389→   389→                File localFolder = new File(shareDir, folderName);
   390→   390→                if (!localFolder.exists()) localFolder.mkdirs();
   391→   391→
   392→   392→                copyFolderFromUri(uri, localFolder);
   393→   393→
   394→   394→                if (fileShareService != null && fileShareService.getServer() != null) {
   395→   395→                    fileShareService.getServer().addSharedFolderPath(localFolder.getAbsolutePath());

   399→   399→                    tvSelectedInfo.setVisibility(View.VISIBLE);

   401→   401→                    Toast.makeText(this, "Folder shared: " + folderName, Toast.LENGTH_SHORT).show();

   403→   403→            } catch (Exception e) {
   404→   404→                Log.e(TAG, "Error sharing folder", e);
   405→   405→                runOnUiThread(() -> Toast.makeText(this, "Error: " + e.getMessage(), Toast.LENGTH_SHORT).show());
   406→   406→            }
   407→   407→        });

   413→   413→     * Correct Android 14 flow:
   414→   414→     * 1. Request FOREGROUND_SERVICE_MEDIA_PROJECTION permission

   416→   416→     * 3. On consent result → start foreground service with result in Intent extras

   418→   418→    private void toggleScreenShare() {
   419→   419→        if (screenSharing) {
   420→   420→            stopScreenSharing();
   421→   421→            return;
   422→   422→        }
   423→   423→        if (!serverRunning) {
   424→   424→            Toast.makeText(this, "Start the server first", Toast.LENGTH_SHORT).show();
   425→   425→            return;
   426→   426→        }
   427→   427→
   428→   428→        // Step 1: Check/request FOREGROUND_SERVICE_MEDIA_PROJECTION
   429→   429→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.UPSIDE_DOWN_CAKE) {
   430→   430→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.FOREGROUND_SERVICE_MEDIA_PROJECTION) != PackageManager.PERMISSION_GRANTED) {
   431→   431→                ActivityCompat.requestPermissions(this,
   432→   432→                    new String[]{Manifest.permission.FOREGROUND_SERVICE_MEDIA_PROJECTION},
   433→   433→                    PERM_REQ_SCREEN);
   434→   434→                return; // result comes back in onRequestPermissionsResult
   435→   435→            }
   436→   436→        }
   437→   437→
   438→   438→        showScreenCaptureDialog();
   439→   439→    }

   448→   448→    /** Step 3: User granted consent → start foreground service WITH the result */
   449→   449→    private void launchScreenShareService(int resultCode, Intent data) {
   450→   450→        try {
   451→   451→            Intent serviceIntent = new Intent(this, ScreenShareService.class);

   506→   506→    private String getLocalIpAddress() {
   507→   507→        try {
   508→   508→            for (NetworkInterface intf : Collections.list(NetworkInterface.getNetworkInterfaces())) {
   509→   509→                for (InetAddress addr : Collections.list(intf.getInetAddresses())) {
   510→   510→                    if (!addr.isLoopbackAddress() && addr.getHostAddress().indexOf(':') < 0) {
   511→   511→                        String ip = addr.getHostAddress();
   512→   512→                        if (ip.startsWith("192.168.") || ip.startsWith("10.") || ip.startsWith("172.")) {
   513→   513→                            return ip;
   514→   514→                        }
   515→   515→                    }
   516→   516→                }
   517→   517→            }
   518→   518→            WifiManager wm = (WifiManager) getApplicationContext().getSystemService(Context.WIFI_SERVICE);
   519→   519→            if (wm != null) {
   520→   520→                WifiInfo wifiInfo = wm.getConnectionInfo();
   521→   521→                int ipInt = wifiInfo.getIpAddress();
   522→   522→                if (ipInt != 0) {
   523→   523→                    return String.format("%d.%d.%d.%d",
   524→   524→                        (ipInt & 0xFF), (ipInt >> 8 & 0xFF),
   525→   525→                        (ipInt >> 16 & 0xFF), (ipInt >> 24 & 0xFF));
   526→   526→                }
   527→   527→            }
   528→   528→        } catch (Exception e) {
   529→   529→            Log.e(TAG, "Error getting IP", e);
   530→   530→        }
   531→   531→        return "0.0.0.0";
   532→   532→    }

   534→   534→    private String getFileName(Uri uri) {
   535→   535→        try {
   536→   536→            if (uri != null && "content".equals(uri.getScheme())) {
   537→   537→                try (android.database.Cursor cursor = getContentResolver().query(uri,
   538→   538→                        new String[]{OpenableColumns.DISPLAY_NAME}, null, null, null)) {
   539→   539→                    if (cursor != null && cursor.moveToFirst()) {
   540→   540→                        int idx = cursor.getColumnIndex(OpenableColumns.DISPLAY_NAME);
   541→   541→                        if (idx >= 0) {
   542→   542→                            String name = cursor.getString(idx);
   543→   543→                            if (!TextUtils.isEmpty(name)) return name;
   544→   544→                        }
   545→   545→                    }
   546→   546→                }
   547→   547→            }
   548→   548→            if (uri != null && uri.getPath() != null) {
   549→   549→                String path = uri.getPath();
   550→   550→                int slash = path.lastIndexOf('/');
   551→   551→                if (slash >= 0) return path.substring(slash + 1);
   552→   552→            }
   553→   553→        } catch (Exception e) {
   554→   554→            Log.e(TAG, "Error getting file name", e);
   555→   555→        }
   556→   556→        return "file_" + System.currentTimeMillis();
   557→   557→    }

   559→   559→    private String getFolderName(Uri uri) {
   560→   560→        try {
   561→   561→            String lastPath = uri.getLastPathSegment();
   562→   562→            if (lastPath != null) {
   563→   563→                int colonIdx = lastPath.lastIndexOf(':');
   564→   564→                if (colonIdx >= 0) lastPath = lastPath.substring(colonIdx + 1);
   565→   565→                if (!TextUtils.isEmpty(lastPath)) return lastPath.replace("/", "_");
   566→   566→            }
   567→   567→        } catch (Exception e) {
   568→   568→            Log.e(TAG, "Error getting folder name", e);
   569→   569→        }
   570→   570→        return "shared_folder_" + System.currentTimeMillis();
   571→   571→    }

   573→   573→    private void copyUriToFile(Uri uri, File destFile) throws Exception {
   574→   574→        InputStream in = getContentResolver().openInputStream(uri);
   575→   575→        if (in == null) throw new Exception("Cannot open file");
   576→   576→        try (OutputStream out = new FileOutputStream(destFile)) {
   577→   577→            byte[] buf = new byte[8192];
   578→   578→            int len;
   579→   579→            while ((len = in.read(buf)) > 0) out.write(buf, 0, len);
   580→   580→        } finally {
   581→   581→            in.close();
   582→   582→        }
   583→   583→    }

   585→   585→    private void copyFolderFromUri(Uri treeUri, File destFolder) throws Exception {
   586→   586→        String treeDocId = DocumentsContract.getTreeDocumentId(treeUri);
   587→   587→        Uri childrenUri = DocumentsContract.buildChildDocumentsUriUsingTree(treeUri, treeDocId);
   588→   588→
   589→   589→        try (android.database.Cursor cursor = getContentResolver().query(childrenUri,
   590→   590→                new String[]{
   591→   591→                    DocumentsContract.Document.COLUMN_DOCUMENT_ID,
   592→   592→                    DocumentsContract.Document.COLUMN_DISPLAY_NAME,
   593→   593→                    DocumentsContract.Document.COLUMN_MIME_TYPE
   594→   594→                }, null, null, null)) {
   595→   595→
   596→   596→            if (cursor != null) {
   597→   597→                while (cursor.moveToNext()) {
   598→   598→                    String docId = cursor.getString(0);
   599→   599→                    String name = cursor.getString(1);
   600→   600→                    String mimeType = cursor.getString(2);
   601→   601→                    Uri docUri = DocumentsContract.buildDocumentUriUsingTree(treeUri, docId);
   602→   602→                    File outFile = new File(destFolder, name);
   603→   603→
   604→   604→                    if (DocumentsContract.Document.MIME_TYPE_DIR.equals(mimeType)) {
   605→   605→                        outFile.mkdirs();
   606→   606→                        copyFolderFromUri(docUri, outFile);
   607→   607→                    } else {
   608→   608→                        copyUriToFile(docUri, outFile);
   609→   609→                    }
   610→   610→                }
   611→   611→            }
   612→   612→        }
   613→   613→    }
