   312‚Üí   312‚Üí   312‚Üí    private void handleFilesSelected(Intent data) {
   313‚Üí   313‚Üí   313‚Üí        List<Uri> uris = new ArrayList<>();
   314‚Üí   314‚Üí   314‚Üí        if (data.getClipData() != null) {
   315‚Üí   315‚Üí   315‚Üí            int count = data.getClipData().getItemCount();
   316‚Üí   316‚Üí   316‚Üí            for (int i = 0; i < count; i++) uris.add(data.getClipData().getItemAt(i).getUri());
   317‚Üí   317‚Üí   317‚Üí        } else if (data.getData() != null) {
   318‚Üí   318‚Üí   318‚Üí            uris.add(data.getData());
   319‚Üí   319‚Üí   319‚Üí        }
   320‚Üí   320‚Üí   320‚Üí        if (uris.isEmpty()) return;
   321‚Üí   321‚Üí   321‚Üí
   322‚Üí   322‚Üí   322‚Üí        Toast.makeText(this, "Copying " + uris.size() + " file(s)...", Toast.LENGTH_SHORT).show();
   323‚Üí   323‚Üí   323‚Üí
   324‚Üí   324‚Üí   324‚Üí        // Copy files on background thread to prevent ANR
   325‚Üí   325‚Üí   325‚Üí        ioExecutor.execute(() -> {
   326‚Üí   326‚Üí   326‚Üí            File shareDir = new File(getExternalFilesDir(null), "shared");
   327‚Üí   327‚Üí   327‚Üí            if (!shareDir.exists()) shareDir.mkdirs();
   328‚Üí   328‚Üí   328‚Üí
   329‚Üí   329‚Üí   329‚Üí            List<String> copiedPaths = new ArrayList<>();
   330‚Üí   330‚Üí   330‚Üí            for (Uri uri : uris) {
   331‚Üí   331‚Üí   331‚Üí                try {
   332‚Üí   332‚Üí   332‚Üí                    String fileName = getFileName(uri);
   333‚Üí   333‚Üí   333‚Üí                    if (fileName == null || fileName.isEmpty()) fileName = "file_" + System.currentTimeMillis();
   334‚Üí   334‚Üí   334‚Üí
   335‚Üí   335‚Üí   335‚Üí                    File destFile = new File(shareDir, fileName);
   336‚Üí   336‚Üí   336‚Üí                    int suffix = 1;
   337‚Üí   337‚Üí   337‚Üí                    while (destFile.exists()) {
   338‚Üí   338‚Üí   338‚Üí                        String base = fileName.contains(".") ? fileName.substring(0, fileName.lastIndexOf('.')) : fileName;
   339‚Üí   339‚Üí   339‚Üí                        String ext = fileName.contains(".") ? fileName.substring(fileName.lastIndexOf('.')) : "";
   340‚Üí   340‚Üí   340‚Üí                        destFile = new File(shareDir, base + " (" + suffix + ")" + ext);
   341‚Üí   341‚Üí   341‚Üí                        suffix++;
   342‚Üí   342‚Üí   342‚Üí                    }
   343‚Üí   343‚Üí   343‚Üí
   344‚Üí   344‚Üí   344‚Üí                    copyUriToFile(uri, destFile);
   345‚Üí   345‚Üí   345‚Üí                    copiedPaths.add(destFile.getAbsolutePath());
   346‚Üí   346‚Üí   346‚Üí                } catch (Exception e) {
   347‚Üí   347‚Üí   347‚Üí                    Log.e(TAG, "Error copying file: " + uri, e);
   348‚Üí   348‚Üí   348‚Üí                    runOnUiThread(() -> Toast.makeText(this, "Error: " + e.getMessage(), Toast.LENGTH_SHORT).show());
   349‚Üí   349‚Üí   349‚Üí                }
   350‚Üí   350‚Üí   350‚Üí            }
   351‚Üí   351‚Üí   351‚Üí
   352‚Üí   352‚Üí   352‚Üí            // Add to server on the IO thread (server uses thread-safe collections)
   353‚Üí   353‚Üí   353‚Üí            if (!copiedPaths.isEmpty() && fileShareService != null && fileShareService.getServer() != null) {
   354‚Üí   354‚Üí   354‚Üí                fileShareService.getServer().addSharedFilePaths(copiedPaths);
   355‚Üí   355‚Üí   355‚Üí            }
   356‚Üí   356‚Üí   356‚Üí
   357‚Üí   357‚Üí   357‚Üí            int count = copiedPaths.size();
   358‚Üí   358‚Üí   358‚Üí            runOnUiThread(() -> {
   359‚Üí   359‚Üí   359‚Üí                tvSelectedInfo.setVisibility(View.VISIBLE);
   360‚Üí   360‚Üí   360‚Üí                tvSelectedInfo.setText(getString(R.string.files_selected, count));
   361‚Üí   361‚Üí   361‚Üí                if (count > 0) {
   362‚Üí   362‚Üí   362‚Üí                    Toast.makeText(this, count + " file(s) shared successfully!", Toast.LENGTH_SHORT).show();
   363‚Üí   363‚Üí   363‚Üí                } else {
   364‚Üí   364‚Üí   364‚Üí                    Toast.makeText(this, "No files were shared", Toast.LENGTH_SHORT).show();
   365‚Üí   365‚Üí   365‚Üí                }
   366‚Üí   366‚Üí   366‚Üí            });
   367‚Üí   367‚Üí   367‚Üí        });
   368‚Üí   368‚Üí   368‚Üí    }

   470‚Üí   470‚Üí   470‚Üí                        try { unbindService(this); } catch (Exception ignored) {}

   476‚Üí   476‚Üí   476‚Üí
   477‚Üí   477‚Üí   477‚Üí            screenSharing = true;
   478‚Üí   478‚Üí   478‚Üí            btnScreenShare.setText(R.string.stop_sharing_screen);

   480‚Üí   480‚Üí   480‚Üí            Toast.makeText(this, "Screen sharing started", Toast.LENGTH_SHORT).show();
   481‚Üí   481‚Üí   481‚Üí

   484‚Üí   484‚Üí   484‚Üí            Toast.makeText(this, "Failed: " + e.getMessage(), Toast.LENGTH_SHORT).show();
   485‚Üí   485‚Üí   485‚Üí        }

   506‚Üí   506‚Üí   506‚Üí    private String getLocalIpAddress() {
   507‚Üí   507‚Üí   507‚Üí        try {
   508‚Üí   508‚Üí   508‚Üí            for (NetworkInterface intf : Collections.list(NetworkInterface.getNetworkInterfaces())) {
   509‚Üí   509‚Üí   509‚Üí                for (InetAddress addr : Collections.list(intf.getInetAddresses())) {
   510‚Üí   510‚Üí   510‚Üí                    if (!addr.isLoopbackAddress() && addr.getHostAddress().indexOf(':') < 0) {
   511‚Üí   511‚Üí   511‚Üí                        String ip = addr.getHostAddress();
   512‚Üí   512‚Üí   512‚Üí                        if (ip.startsWith("192.168.") || ip.startsWith("10.") || ip.startsWith("172.")) {
   513‚Üí   513‚Üí   513‚Üí                            return ip;
   514‚Üí   514‚Üí   514‚Üí                        }
   515‚Üí   515‚Üí   515‚Üí                    }
   516‚Üí   516‚Üí   516‚Üí                }
   517‚Üí   517‚Üí   517‚Üí            }
   518‚Üí   518‚Üí   518‚Üí            WifiManager wm = (WifiManager) getApplicationContext().getSystemService(Context.WIFI_SERVICE);
   519‚Üí   519‚Üí   519‚Üí            if (wm != null) {
   520‚Üí   520‚Üí   520‚Üí                WifiInfo wifiInfo = wm.getConnectionInfo();
   521‚Üí   521‚Üí   521‚Üí                int ipInt = wifiInfo.getIpAddress();
   522‚Üí   522‚Üí   522‚Üí                if (ipInt != 0) {
   523‚Üí   523‚Üí   523‚Üí                    return String.format("%d.%d.%d.%d",
   524‚Üí   524‚Üí   524‚Üí                        (ipInt & 0xFF), (ipInt >> 8 & 0xFF),
   525‚Üí   525‚Üí   525‚Üí                        (ipInt >> 16 & 0xFF), (ipInt >> 24 & 0xFF));
   526‚Üí   526‚Üí   526‚Üí                }
   527‚Üí   527‚Üí   527‚Üí            }
   528‚Üí   528‚Üí   528‚Üí        } catch (Exception e) {
   529‚Üí   529‚Üí   529‚Üí            Log.e(TAG, "Error getting IP", e);
   530‚Üí   530‚Üí   530‚Üí        }
   531‚Üí   531‚Üí   531‚Üí        return "0.0.0.0";
   532‚Üí   532‚Üí   532‚Üí    }

   534‚Üí   534‚Üí   534‚Üí    private String getFileName(Uri uri) {
   535‚Üí   535‚Üí   535‚Üí        try {
   536‚Üí   536‚Üí   536‚Üí            if (uri != null && "content".equals(uri.getScheme())) {
   537‚Üí   537‚Üí   537‚Üí                try (android.database.Cursor cursor = getContentResolver().query(uri,
   538‚Üí   538‚Üí   538‚Üí                        new String[]{OpenableColumns.DISPLAY_NAME}, null, null, null)) {
   539‚Üí   539‚Üí   539‚Üí                    if (cursor != null && cursor.moveToFirst()) {
   540‚Üí   540‚Üí   540‚Üí                        int idx = cursor.getColumnIndex(OpenableColumns.DISPLAY_NAME);
   541‚Üí   541‚Üí   541‚Üí                        if (idx >= 0) {
   542‚Üí   542‚Üí   542‚Üí                            String name = cursor.getString(idx);
   543‚Üí   543‚Üí   543‚Üí                            if (!TextUtils.isEmpty(name)) return name;
   544‚Üí   544‚Üí   544‚Üí                        }

   546‚Üí   546‚Üí   546‚Üí                }

   549‚Üí   549‚Üí   549‚Üí                String path = uri.getPath();
   550‚Üí   550‚Üí   550‚Üí                int slash = path.lastIndexOf('/');
   551‚Üí   551‚Üí   551‚Üí                if (slash >= 0) return path.substring(slash + 1);
   552‚Üí   552‚Üí   552‚Üí            }
   553‚Üí   553‚Üí   553‚Üí        } catch (Exception e) {
   554‚Üí   554‚Üí   554‚Üí            Log.e(TAG, "Error getting file name", e);
   555‚Üí   555‚Üí   555‚Üí        }

   560‚Üí   560‚Üí   560‚Üí        try {
   561‚Üí   561‚Üí   561‚Üí            String lastPath = uri.getLastPathSegment();

   573‚Üí   573‚Üí   573‚Üí    private void copyUriToFile(Uri uri, File destFile) throws Exception {
   574‚Üí   574‚Üí   574‚Üí        InputStream in = getContentResolver().openInputStream(uri);
   575‚Üí   575‚Üí   575‚Üí        if (in == null) throw new Exception("Cannot open file");
   576‚Üí   576‚Üí   576‚Üí        try (OutputStream out = new FileOutputStream(destFile)) {
   577‚Üí   577‚Üí   577‚Üí            byte[] buf = new byte[8192];

   580‚Üí   580‚Üí   580‚Üí        } finally {
   581‚Üí   581‚Üí   581‚Üí            in.close();

   585‚Üí   585‚Üí   585‚Üí    private void copyFolderFromUri(Uri treeUri, File destFolder) throws Exception {
   586‚Üí   586‚Üí   586‚Üí        String treeDocId = DocumentsContract.getTreeDocumentId(treeUri);
   587‚Üí   587‚Üí   587‚Üí        Uri childrenUri = DocumentsContract.buildChildDocumentsUriUsingTree(treeUri, treeDocId);
   588‚Üí   588‚Üí   588‚Üí
   589‚Üí   589‚Üí   589‚Üí        try (android.database.Cursor cursor = getContentResolver().query(childrenUri,
   590‚Üí   590‚Üí   590‚Üí                new String[]{
   591‚Üí   591‚Üí   591‚Üí                    DocumentsContract.Document.COLUMN_DOCUMENT_ID,
   592‚Üí   592‚Üí   592‚Üí                    DocumentsContract.Document.COLUMN_DISPLAY_NAME,
   593‚Üí   593‚Üí   593‚Üí                    DocumentsContract.Document.COLUMN_MIME_TYPE
   594‚Üí   594‚Üí   594‚Üí                }, null, null, null)) {
   595‚Üí   595‚Üí   595‚Üí
   596‚Üí   596‚Üí   596‚Üí            if (cursor != null) {
   597‚Üí   597‚Üí   597‚Üí                while (cursor.moveToNext()) {
   598‚Üí   598‚Üí   598‚Üí                    String docId = cursor.getString(0);
   599‚Üí   599‚Üí   599‚Üí                    String name = cursor.getString(1);
   600‚Üí   600‚Üí   600‚Üí                    String mimeType = cursor.getString(2);
   601‚Üí   601‚Üí   601‚Üí                    Uri docUri = DocumentsContract.buildDocumentUriUsingTree(treeUri, docId);
   602‚Üí   602‚Üí   602‚Üí                    File outFile = new File(destFolder, name);
   603‚Üí   603‚Üí   603‚Üí
   604‚Üí   604‚Üí   604‚Üí                    if (DocumentsContract.Document.MIME_TYPE_DIR.equals(mimeType)) {
   605‚Üí   605‚Üí   605‚Üí                        outFile.mkdirs();
   606‚Üí   606‚Üí   606‚Üí                        copyFolderFromUri(docUri, outFile);
   607‚Üí   607‚Üí   607‚Üí                    } else {
   608‚Üí   608‚Üí   608‚Üí                        copyUriToFile(docUri, outFile);
   609‚Üí   609‚Üí   609‚Üí                    }
   610‚Üí   610‚Üí   610‚Üí                }
   611‚Üí   611‚Üí   611‚Üí            }
   612‚Üí   612‚Üí   612‚Üí        }
   613‚Üí   613‚Üí   613‚Üí    }
