   167→   167→    private void requestAllPermissions() {
   168→   168→        List<String> needed = new ArrayList<>();
   169→   169→
   170→   170→        if (ContextCompat.checkSelfPermission(this, Manifest.permission.POST_NOTIFICATIONS) != PackageManager.PERMISSION_GRANTED) {
   171→   171→            needed.add(Manifest.permission.POST_NOTIFICATIONS);
   172→   172→        }
   173→   173→
   174→   174→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
   175→   175→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_IMAGES) != PackageManager.PERMISSION_GRANTED)
   176→   176→                needed.add(Manifest.permission.READ_MEDIA_IMAGES);
   177→   177→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_VIDEO) != PackageManager.PERMISSION_GRANTED)
   178→   178→                needed.add(Manifest.permission.READ_MEDIA_VIDEO);
   179→   179→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_AUDIO) != PackageManager.PERMISSION_GRANTED)
   180→   180→                needed.add(Manifest.permission.READ_MEDIA_AUDIO);
   181→   181→        } else {
   182→   182→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_EXTERNAL_STORAGE) != PackageManager.PERMISSION_GRANTED)
   183→   183→                needed.add(Manifest.permission.READ_EXTERNAL_STORAGE);
   184→   184→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.WRITE_EXTERNAL_STORAGE) != PackageManager.PERMISSION_GRANTED)
   185→   185→                needed.add(Manifest.permission.WRITE_EXTERNAL_STORAGE);
   186→   186→        }
   187→   187→
   188→   188→        if (!needed.isEmpty()) {
   189→   189→            ActivityCompat.requestPermissions(this, needed.toArray(new String[0]), PERM_REQ_GENERAL);
   190→   190→        }
   191→   191→
   192→   192→        // MANAGE_EXTERNAL_STORAGE for full folder access (Android 11+)
   193→   193→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R && !Environment.isExternalStorageManager()) {
   194→   194→            try {
   195→   195→                startActivity(new Intent(Settings.ACTION_MANAGE_APP_ALL_FILES_ACCESS_PERMISSION,
   196→   196→                    Uri.parse("package:" + getPackageName())));
   197→   197→            } catch (Exception e) {
   198→   198→                startActivity(new Intent(Settings.ACTION_MANAGE_ALL_FILES_ACCESS_PERMISSION));
   199→   199→            }
   200→   200→        }
   201→   201→    }

   316→   316→    private void handleFilesSelected(Intent data) {
   317→   317→        List<Uri> uris = new ArrayList<>();
   318→   318→        if (data.getClipData() != null) {
   319→   319→            int count = data.getClipData().getItemCount();
   320→   320→            for (int i = 0; i < count; i++) uris.add(data.getClipData().getItemAt(i).getUri());
   321→   321→        } else if (data.getData() != null) {
   322→   322→            uris.add(data.getData());
   323→   323→        }
   324→   324→        if (uris.isEmpty()) return;
   325→   325→
   326→   326→        Toast.makeText(this, "Copying " + uris.size() + " file(s)...", Toast.LENGTH_SHORT).show();
   327→   327→
   328→   328→        // Copy files on background thread to prevent ANR
   329→   329→        ioExecutor.execute(() -> {
   330→   330→            File shareDir = new File(getExternalFilesDir(null), "shared");
   331→   331→            if (!shareDir.exists()) shareDir.mkdirs();
   332→   332→
   333→   333→            List<String> copiedPaths = new ArrayList<>();
   334→   334→            for (Uri uri : uris) {
   335→   335→                try {
   336→   336→                    String fileName = getFileName(uri);
   337→   337→                    if (fileName == null || fileName.isEmpty()) fileName = "file_" + System.currentTimeMillis();
   338→   338→
   339→   339→                    File destFile = new File(shareDir, fileName);
   340→   340→                    int suffix = 1;
   341→   341→                    while (destFile.exists()) {
   342→   342→                        String base = fileName.contains(".") ? fileName.substring(0, fileName.lastIndexOf('.')) : fileName;
   343→   343→                        String ext = fileName.contains(".") ? fileName.substring(fileName.lastIndexOf('.')) : "";
   344→   344→                        destFile = new File(shareDir, base + " (" + suffix + ")" + ext);
   345→   345→                        suffix++;
   346→   346→                    }
   347→   347→
   348→   348→                    copyUriToFile(uri, destFile);
   349→   349→                    copiedPaths.add(destFile.getAbsolutePath());
   350→   350→                } catch (Exception e) {
   351→   351→                    Log.e(TAG, "Error copying file: " + uri, e);
   352→   352→                    runOnUiThread(() -> Toast.makeText(this, "Error: " + e.getMessage(), Toast.LENGTH_SHORT).show());
   353→   353→                }
   354→   354→            }
   355→   355→
   356→   356→            // Add to server on the IO thread (server uses thread-safe collections)
   357→   357→            if (!copiedPaths.isEmpty() && fileShareService != null && fileShareService.getServer() != null) {
   358→   358→                fileShareService.getServer().addSharedFilePaths(copiedPaths);
   359→   359→            }
   360→   360→
   361→   361→            int count = copiedPaths.size();
   362→   362→            runOnUiThread(() -> {
   363→   363→                tvSelectedInfo.setVisibility(View.VISIBLE);
   364→   364→                tvSelectedInfo.setText(getString(R.string.files_selected, count));
   365→   365→                if (count > 0) {
   366→   366→                    Toast.makeText(this, count + " file(s) shared successfully!", Toast.LENGTH_SHORT).show();
   367→   367→                } else {
   368→   368→                    Toast.makeText(this, "No files were shared", Toast.LENGTH_SHORT).show();
   369→   369→                }
   370→   370→            });
   371→   371→        });
   372→   372→    }

   384→   384→    private void handleFolderSelected(Uri uri) {
   385→   385→        Toast.makeText(this, "Copying folder...", Toast.LENGTH_SHORT).show();
   386→   386→
   387→   387→        ioExecutor.execute(() -> {
   388→   388→            try {
   389→   389→                File shareDir = new File(getExternalFilesDir(null), "shared");
   390→   390→                if (!shareDir.exists()) shareDir.mkdirs();
   391→   391→
   392→   392→                String folderName = getFolderName(uri);
   393→   393→                File localFolder = new File(shareDir, folderName);
   394→   394→                if (!localFolder.exists()) localFolder.mkdirs();
   395→   395→
   396→   396→                copyFolderFromUri(uri, localFolder);
   397→   397→
   398→   398→                if (fileShareService != null && fileShareService.getServer() != null) {
   399→   399→                    fileShareService.getServer().addSharedFolderPath(localFolder.getAbsolutePath());
   400→   400→                }
   401→   401→
   402→   402→                runOnUiThread(() -> {
   403→   403→                    tvSelectedInfo.setVisibility(View.VISIBLE);
   404→   404→                    tvSelectedInfo.setText("Folder shared: " + folderName);
   405→   405→                    Toast.makeText(this, "Folder shared: " + folderName, Toast.LENGTH_SHORT).show();
   406→   406→                });
   407→   407→            } catch (Exception e) {
   408→   408→                Log.e(TAG, "Error sharing folder", e);
   409→   409→                runOnUiThread(() -> Toast.makeText(this, "Error: " + e.getMessage(), Toast.LENGTH_SHORT).show());
   410→   410→            }
   411→   411→        });

   417→   417→     * Correct Android 14 flow:
   418→   418→     * 1. Request FOREGROUND_SERVICE_MEDIA_PROJECTION permission

   420→   420→     * 3. On consent result → start foreground service with result in Intent extras

   422→   422→    private void toggleScreenShare() {
   423→   423→        if (screenSharing) {
   424→   424→            stopScreenSharing();
   425→   425→            return;
   426→   426→        }
   427→   427→        if (!serverRunning) {
   428→   428→            Toast.makeText(this, "Start the server first", Toast.LENGTH_SHORT).show();
   429→   429→            return;
   430→   430→        }
   431→   431→
   432→   432→        // Step 1: Check/request FOREGROUND_SERVICE_MEDIA_PROJECTION
   433→   433→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.UPSIDE_DOWN_CAKE) {
   434→   434→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.FOREGROUND_SERVICE_MEDIA_PROJECTION) != PackageManager.PERMISSION_GRANTED) {
   435→   435→                ActivityCompat.requestPermissions(this,
   436→   436→                    new String[]{Manifest.permission.FOREGROUND_SERVICE_MEDIA_PROJECTION},
   437→   437→                    PERM_REQ_SCREEN);
   438→   438→                return; // result comes back in onRequestPermissionsResult
   439→   439→            }
   440→   440→        }
   441→   441→
   442→   442→        showScreenCaptureDialog();
   443→   443→    }

   453→   453→    private void launchScreenShareService(int resultCode, Intent data) {
   454→   454→        try {
   455→   455→            Intent serviceIntent = new Intent(this, ScreenShareService.class);
   456→   456→            serviceIntent.putExtra(ScreenShareService.EXTRA_RESULT_CODE, resultCode);
   457→   457→            serviceIntent.putExtra(ScreenShareService.EXTRA_PROJECTION_DATA, data);
   458→   458→
   459→   459→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
   460→   460→                startForegroundService(serviceIntent);
   461→   461→            } else {
   462→   462→                startService(serviceIntent);
   463→   463→            }
   464→   464→
   465→   465→            // Connect the screen service to the file server for frame delivery
   466→   466→            if (fileShareService != null && fileShareService.getServer() != null) {
   467→   467→                fileShareService.getServer().setScreenSharingActive(true);
   468→   468→                // Bind briefly to wire up the server reference
   469→   469→                bindService(new Intent(this, ScreenShareService.class), new ServiceConnection() {
   470→   470→                    @Override
   471→   471→                    public void onServiceConnected(ComponentName name, IBinder service) {
   472→   472→                        ScreenShareService screenSvc = ((ScreenShareService.LocalBinder) service).getService();
   473→   473→                        screenSvc.setFileShareServer(fileShareService.getServer());
   474→   474→                        try { unbindService(this); } catch (Exception ignored) {}
   475→   475→                    }
   476→   476→                    @Override
   477→   477→                    public void onServiceDisconnected(ComponentName name) {}
   478→   478→                }, Context.BIND_AUTO_CREATE);
   479→   479→            }
   480→   480→
   481→   481→            screenSharing = true;
   482→   482→            btnScreenShare.setText(R.string.stop_sharing_screen);
   483→   483→            btnScreenShare.setBackgroundColor(getColor(R.color.red));
   484→   484→            Toast.makeText(this, "Screen sharing started", Toast.LENGTH_SHORT).show();
   485→   485→
   486→   486→        } catch (Exception e) {
   487→   487→            Log.e(TAG, "Failed to start screen share", e);
   488→   488→            Toast.makeText(this, "Failed: " + e.getMessage(), Toast.LENGTH_SHORT).show();
   489→   489→        }
   490→   490→    }

   536→   536→    private String getLocalIpAddress() {
   537→   537→        try {
   538→   538→            for (NetworkInterface intf : Collections.list(NetworkInterface.getNetworkInterfaces())) {
   539→   539→                for (InetAddress addr : Collections.list(intf.getInetAddresses())) {
   540→   540→                    if (!addr.isLoopbackAddress() && addr.getHostAddress().indexOf(':') < 0) {
   541→   541→                        String ip = addr.getHostAddress();
   542→   542→                        if (ip.startsWith("192.168.") || ip.startsWith("10.") || ip.startsWith("172.")) {
   543→   543→                            return ip;
   544→   544→                        }
   545→   545→                    }
   546→   546→                }
   547→   547→            }
   548→   548→            WifiManager wm = (WifiManager) getApplicationContext().getSystemService(Context.WIFI_SERVICE);
   549→   549→            if (wm != null) {
   550→   550→                WifiInfo wifiInfo = wm.getConnectionInfo();
   551→   551→                int ipInt = wifiInfo.getIpAddress();
   552→   552→                if (ipInt != 0) {
   553→   553→                    return String.format("%d.%d.%d.%d",
   554→   554→                        (ipInt & 0xFF), (ipInt >> 8 & 0xFF),
   555→   555→                        (ipInt >> 16 & 0xFF), (ipInt >> 24 & 0xFF));
   556→   556→                }
   557→   557→            }
   558→   558→        } catch (Exception e) {
   559→   559→            Log.e(TAG, "Error getting IP", e);
   560→   560→        }
   561→   561→        return "0.0.0.0";
   562→   562→    }

   564→   564→    private String getFileName(Uri uri) {
   565→   565→        try {
   566→   566→            if (uri != null && "content".equals(uri.getScheme())) {
   567→   567→                try (android.database.Cursor cursor = getContentResolver().query(uri,
   568→   568→                        new String[]{OpenableColumns.DISPLAY_NAME}, null, null, null)) {
   569→   569→                    if (cursor != null && cursor.moveToFirst()) {
   570→   570→                        int idx = cursor.getColumnIndex(OpenableColumns.DISPLAY_NAME);
   571→   571→                        if (idx >= 0) {
   572→   572→                            String name = cursor.getString(idx);
   573→   573→                            if (!TextUtils.isEmpty(name)) return name;
   574→   574→                        }
   575→   575→                    }
   576→   576→                }
   577→   577→            }
   578→   578→            if (uri != null && uri.getPath() != null) {
   579→   579→                String path = uri.getPath();
   580→   580→                int slash = path.lastIndexOf('/');
   581→   581→                if (slash >= 0) return path.substring(slash + 1);
   582→   582→            }
   583→   583→        } catch (Exception e) {
   584→   584→            Log.e(TAG, "Error getting file name", e);
   585→   585→        }
   586→   586→        return "file_" + System.currentTimeMillis();
   587→   587→    }

   589→   589→    private String getFolderName(Uri uri) {
   590→   590→        try {
   591→   591→            String lastPath = uri.getLastPathSegment();
   592→   592→            if (lastPath != null) {
   593→   593→                int colonIdx = lastPath.lastIndexOf(':');
   594→   594→                if (colonIdx >= 0) lastPath = lastPath.substring(colonIdx + 1);
   595→   595→                if (!TextUtils.isEmpty(lastPath)) return lastPath.replace("/", "_");
   596→   596→            }
   597→   597→        } catch (Exception e) {
   598→   598→            Log.e(TAG, "Error getting folder name", e);
   599→   599→        }
   600→   600→        return "shared_folder_" + System.currentTimeMillis();
   601→   601→    }

   603→   603→    private void copyUriToFile(Uri uri, File destFile) throws Exception {
   604→   604→        InputStream in = getContentResolver().openInputStream(uri);
   605→   605→        if (in == null) throw new Exception("Cannot open file");
   606→   606→        try (OutputStream out = new FileOutputStream(destFile)) {
   607→   607→            byte[] buf = new byte[8192];
   608→   608→            int len;
   609→   609→            while ((len = in.read(buf)) > 0) out.write(buf, 0, len);
   610→   610→        } finally {
   611→   611→            in.close();
   612→   612→        }
   613→   613→    }

   615→   615→    private void copyFolderFromUri(Uri treeUri, File destFolder) throws Exception {
   616→   616→        String treeDocId = DocumentsContract.getTreeDocumentId(treeUri);
   617→   617→        Uri childrenUri = DocumentsContract.buildChildDocumentsUriUsingTree(treeUri, treeDocId);
   618→   618→
   619→   619→        try (android.database.Cursor cursor = getContentResolver().query(childrenUri,
   620→   620→                new String[]{
   621→   621→                    DocumentsContract.Document.COLUMN_DOCUMENT_ID,
   622→   622→                    DocumentsContract.Document.COLUMN_DISPLAY_NAME,
   623→   623→                    DocumentsContract.Document.COLUMN_MIME_TYPE
   624→   624→                }, null, null, null)) {
   625→   625→
   626→   626→            if (cursor != null) {
   627→   627→                while (cursor.moveToNext()) {
   628→   628→                    String docId = cursor.getString(0);
   629→   629→                    String name = cursor.getString(1);
   630→   630→                    String mimeType = cursor.getString(2);
   631→   631→                    Uri docUri = DocumentsContract.buildDocumentUriUsingTree(treeUri, docId);
   632→   632→                    File outFile = new File(destFolder, name);
   633→   633→
   634→   634→                    if (DocumentsContract.Document.MIME_TYPE_DIR.equals(mimeType)) {
   635→   635→                        outFile.mkdirs();
   636→   636→                        copyFolderFromUri(docUri, outFile);
   637→   637→                    } else {
   638→   638→                        copyUriToFile(docUri, outFile);
   639→   639→                    }
   640→   640→                }
   641→   641→            }
   642→   642→        }
   643→   643→    }
