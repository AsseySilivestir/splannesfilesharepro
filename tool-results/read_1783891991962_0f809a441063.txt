   167→    private void requestAllPermissions() {
   168→        List<String> needed = new ArrayList<>();
   169→
   170→        if (ContextCompat.checkSelfPermission(this, Manifest.permission.POST_NOTIFICATIONS) != PackageManager.PERMISSION_GRANTED) {
   171→            needed.add(Manifest.permission.POST_NOTIFICATIONS);
   172→        }
   173→
   174→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
   175→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_IMAGES) != PackageManager.PERMISSION_GRANTED)
   176→                needed.add(Manifest.permission.READ_MEDIA_IMAGES);
   177→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_VIDEO) != PackageManager.PERMISSION_GRANTED)
   178→                needed.add(Manifest.permission.READ_MEDIA_VIDEO);
   179→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_MEDIA_AUDIO) != PackageManager.PERMISSION_GRANTED)
   180→                needed.add(Manifest.permission.READ_MEDIA_AUDIO);
   181→        } else {
   182→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.READ_EXTERNAL_STORAGE) != PackageManager.PERMISSION_GRANTED)
   183→                needed.add(Manifest.permission.READ_EXTERNAL_STORAGE);
   184→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.WRITE_EXTERNAL_STORAGE) != PackageManager.PERMISSION_GRANTED)
   185→                needed.add(Manifest.permission.WRITE_EXTERNAL_STORAGE);
   186→        }
   187→
   188→        if (!needed.isEmpty()) {
   189→            ActivityCompat.requestPermissions(this, needed.toArray(new String[0]), PERM_REQ_GENERAL);
   190→        }
   191→
   192→        // MANAGE_EXTERNAL_STORAGE for full folder access (Android 11+)
   193→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R && !Environment.isExternalStorageManager()) {
   194→            try {
   195→                startActivity(new Intent(Settings.ACTION_MANAGE_APP_ALL_FILES_ACCESS_PERMISSION,
   196→                    Uri.parse("package:" + getPackageName())));
   197→            } catch (Exception e) {
   198→                startActivity(new Intent(Settings.ACTION_MANAGE_ALL_FILES_ACCESS_PERMISSION));
   199→            }
   200→        }
   201→    }

   204→    public void onRequestPermissionsResult(int requestCode, @NonNull String[] permissions, @NonNull int[] grantResults) {
   205→        super.onRequestPermissionsResult(requestCode, permissions, grantResults);
   206→        if (requestCode == PERM_REQ_SCREEN) {

   213→        }

   216→    private boolean allGranted(int[] results) {
   217→        for (int r : results) if (r != PackageManager.PERMISSION_GRANTED) return false;
   218→        return true;
   219→    }

   316→    private void handleFilesSelected(Intent data) {
   317→        List<Uri> uris = new ArrayList<>();
   318→        if (data.getClipData() != null) {
   319→            int count = data.getClipData().getItemCount();
   320→            for (int i = 0; i < count; i++) uris.add(data.getClipData().getItemAt(i).getUri());
   321→        } else if (data.getData() != null) {
   322→            uris.add(data.getData());
   323→        }
   324→        if (uris.isEmpty()) return;
   325→
   326→        Toast.makeText(this, "Copying " + uris.size() + " file(s)...", Toast.LENGTH_SHORT).show();
   327→
   328→        // Copy files on background thread to prevent ANR
   329→        ioExecutor.execute(() -> {
   330→            File shareDir = new File(getExternalFilesDir(null), "shared");
   331→            if (!shareDir.exists()) shareDir.mkdirs();
   332→
   333→            List<String> copiedPaths = new ArrayList<>();
   334→            for (Uri uri : uris) {
   335→                try {
   336→                    String fileName = getFileName(uri);
   337→                    if (fileName == null || fileName.isEmpty()) fileName = "file_" + System.currentTimeMillis();
   338→
   339→                    File destFile = new File(shareDir, fileName);
   340→                    int suffix = 1;
   341→                    while (destFile.exists()) {
   342→                        String base = fileName.contains(".") ? fileName.substring(0, fileName.lastIndexOf('.')) : fileName;
   343→                        String ext = fileName.contains(".") ? fileName.substring(fileName.lastIndexOf('.')) : "";
   344→                        destFile = new File(shareDir, base + " (" + suffix + ")" + ext);
   345→                        suffix++;
   346→                    }
   347→
   348→                    copyUriToFile(uri, destFile);
   349→                    copiedPaths.add(destFile.getAbsolutePath());
   350→                } catch (Exception e) {
   351→                    Log.e(TAG, "Error copying file: " + uri, e);
   352→                    runOnUiThread(() -> Toast.makeText(this, "Error: " + e.getMessage(), Toast.LENGTH_SHORT).show());
   353→                }
   354→            }
   355→
   356→            // Add to server on the IO thread (server uses thread-safe collections)
   357→            if (!copiedPaths.isEmpty() && fileShareService != null && fileShareService.getServer() != null) {
   358→                fileShareService.getServer().addSharedFilePaths(copiedPaths);
   359→            }
   360→
   361→            int count = copiedPaths.size();
   362→            runOnUiThread(() -> {
   363→                tvSelectedInfo.setVisibility(View.VISIBLE);
   364→                tvSelectedInfo.setText(getString(R.string.files_selected, count));
   365→                if (count > 0) {
   366→                    Toast.makeText(this, count + " file(s) shared successfully!", Toast.LENGTH_SHORT).show();
   367→                } else {
   368→                    Toast.makeText(this, "No files were shared", Toast.LENGTH_SHORT).show();
   369→                }
   370→            });
   371→        });
   372→    }

   384→    private void handleFolderSelected(Uri uri) {

   386→
   387→        ioExecutor.execute(() -> {
   388→            try {
   389→                File shareDir = new File(getExternalFilesDir(null), "shared");

   391→
   392→                String folderName = getFolderName(uri);
   393→                File localFolder = new File(shareDir, folderName);
   394→                if (!localFolder.exists()) localFolder.mkdirs();
   395→
   396→                copyFolderFromUri(uri, localFolder);
   397→
   398→                if (fileShareService != null && fileShareService.getServer() != null) {
   399→                    fileShareService.getServer().addSharedFolderPath(localFolder.getAbsolutePath());
   400→                }
   401→
   402→                runOnUiThread(() -> {
   403→                    tvSelectedInfo.setVisibility(View.VISIBLE);
   404→                    tvSelectedInfo.setText("Folder shared: " + folderName);
   405→                    Toast.makeText(this, "Folder shared: " + folderName, Toast.LENGTH_SHORT).show();
   406→                });
   407→            } catch (Exception e) {
   408→                Log.e(TAG, "Error sharing folder", e);
   409→                runOnUiThread(() -> Toast.makeText(this, "Error: " + e.getMessage(), Toast.LENGTH_SHORT).show());
   410→            }
   411→        });

   414→    // ===================== Screen sharing (fixed Android 14 flow) =====================
   415→
   416→    /**
   417→     * Correct Android 14 flow:
   418→     * 1. Request FOREGROUND_SERVICE_MEDIA_PROJECTION permission
   419→     * 2. Show system MediaProjection consent dialog
   420→     * 3. On consent result → start foreground service with result in Intent extras
   421→     */
   422→    private void toggleScreenShare() {
   423→        if (screenSharing) {
   424→            stopScreenSharing();
   425→            return;
   426→        }
   427→        if (!serverRunning) {
   428→            Toast.makeText(this, "Start the server first", Toast.LENGTH_SHORT).show();
   429→            return;
   430→        }
   431→
   432→        // Step 1: Check/request FOREGROUND_SERVICE_MEDIA_PROJECTION
   433→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.UPSIDE_DOWN_CAKE) {
   434→            if (ContextCompat.checkSelfPermission(this, Manifest.permission.FOREGROUND_SERVICE_MEDIA_PROJECTION) != PackageManager.PERMISSION_GRANTED) {
   435→                ActivityCompat.requestPermissions(this,
   436→                    new String[]{Manifest.permission.FOREGROUND_SERVICE_MEDIA_PROJECTION},
   437→                    PERM_REQ_SCREEN);
   438→                return; // result comes back in onRequestPermissionsResult
   439→            }
   440→        }
   441→
   442→        showScreenCaptureDialog();
   443→    }
   444→
   445→    /** Step 2: Show the system screen capture consent dialog */
   446→    private void showScreenCaptureDialog() {
   447→        android.media.projection.MediaProjectionManager pm =
   448→            (android.media.projection.MediaProjectionManager) getSystemService(Context.MEDIA_PROJECTION_SERVICE);
   449→        screenCaptureLauncher.launch(pm.createScreenCaptureIntent());
   450→    }

   452→    /** Step 3: User granted consent → start foreground service WITH the result */
   453→    private void launchScreenShareService(int resultCode, Intent data) {
   454→        try {
   455→            Intent serviceIntent = new Intent(this, ScreenShareService.class);
   456→            serviceIntent.putExtra(ScreenShareService.EXTRA_RESULT_CODE, resultCode);
   457→            serviceIntent.putExtra(ScreenShareService.EXTRA_PROJECTION_DATA, data);
   458→
   459→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
   460→                startForegroundService(serviceIntent);
   461→            } else {
   462→                startService(serviceIntent);
   463→            }
   464→
   465→            // Connect the screen service to the file server for frame delivery
   466→            // CRITICAL: setScreenSharingActive(true) must happen AFTER setFileShareServer(),
   467→            // otherwise the MJPEG stream starts but no frames arrive (fileShareServer is null
   468→            // in ScreenShareService when it tries to pushScreenFrame).
   469→            if (fileShareService != null && fileShareService.getServer() != null) {
   470→                bindService(new Intent(this, ScreenShareService.class), new ServiceConnection() {
   471→                    @Override
   472→                    public void onServiceConnected(ComponentName name, IBinder service) {
   473→                        ScreenShareService screenSvc = ((ScreenShareService.LocalBinder) service).getService();
   474→                        FileShareServer server = fileShareService.getServer();
   475→                        // Wire server into screen service FIRST
   476→                        screenSvc.setFileShareServer(server);
   477→                        // THEN activate the stream so browser sees frames immediately
   478→                        server.setScreenSharingActive(true);
   479→                        try { unbindService(this); } catch (Exception ignored) {}
   480→                    }
   481→                    @Override
   482→                    public void onServiceDisconnected(ComponentName name) {}
   483→                }, Context.BIND_AUTO_CREATE);
   484→            }
   485→
   486→            screenSharing = true;
   487→            btnScreenShare.setText(R.string.stop_sharing_screen);
   488→            btnScreenShare.setBackgroundColor(getColor(R.color.red));
   489→            Toast.makeText(this, "Screen sharing started", Toast.LENGTH_SHORT).show();
   490→
   491→        } catch (Exception e) {
   492→            Log.e(TAG, "Failed to start screen share", e);
   493→            Toast.makeText(this, "Failed: " + e.getMessage(), Toast.LENGTH_SHORT).show();
   494→        }
   495→    }
   496→
   497→    private void stopScreenSharing() {
   498→        try {
   499→            stopService(new Intent(this, ScreenShareService.class));
   500→        } catch (Exception ignored) {}
   501→
   502→        if (fileShareService != null && fileShareService.getServer() != null) {
   503→            fileShareService.getServer().setScreenSharingActive(false);
   504→            fileShareService.getServer().setScreenCaptureCallback(null);
   505→        }
   506→
   507→        screenSharing = false;
   508→        btnScreenShare.setText(R.string.share_screen);
   509→        btnScreenShare.setBackgroundColor(getColor(R.color.accent));
   510→        Toast.makeText(this, "Screen sharing stopped", Toast.LENGTH_SHORT).show();
   511→    }

   524→                    Toast.makeText(this, files.length + " file(s) in Downloads/SplannesFileShare", Toast.LENGTH_LONG).show();
   525→                    // Try to open with a file manager
   526→                    Intent intent = new Intent(Intent.ACTION_VIEW);
   527→                    intent.setDataAndType(Uri.fromFile(shareDir), "resource/folder");
   528→                    intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
   529→                    try { startActivity(intent); } catch (Exception e) {
   530→                        Toast.makeText(this, "Open Downloads/SplannesFileShare in your file manager", Toast.LENGTH_LONG).show();
   531→                    }
   532→                    return;

   543→    private String getLocalIpAddress() {
   544→        try {
   545→            for (NetworkInterface intf : Collections.list(NetworkInterface.getNetworkInterfaces())) {
   546→                for (InetAddress addr : Collections.list(intf.getInetAddresses())) {
   547→                    if (!addr.isLoopbackAddress() && addr.getHostAddress().indexOf(':') < 0) {
   548→                        String ip = addr.getHostAddress();
   549→                        if (ip.startsWith("192.168.") || ip.startsWith("10.") || ip.startsWith("172.")) {
   550→                            return ip;
   551→                        }
   552→                    }
   553→                }
   554→            }
   555→            WifiManager wm = (WifiManager) getApplicationContext().getSystemService(Context.WIFI_SERVICE);
   556→            if (wm != null) {
   557→                WifiInfo wifiInfo = wm.getConnectionInfo();
   558→                int ipInt = wifiInfo.getIpAddress();
   559→                if (ipInt != 0) {
   560→                    return String.format("%d.%d.%d.%d",
   561→                        (ipInt & 0xFF), (ipInt >> 8 & 0xFF),
   562→                        (ipInt >> 16 & 0xFF), (ipInt >> 24 & 0xFF));
   563→                }
   564→            }
   565→        } catch (Exception e) {
   566→            Log.e(TAG, "Error getting IP", e);
   567→        }
   568→        return "0.0.0.0";
   569→    }

   571→    private String getFileName(Uri uri) {
   572→        try {
   573→            if (uri != null && "content".equals(uri.getScheme())) {
   574→                try (android.database.Cursor cursor = getContentResolver().query(uri,
   575→                        new String[]{OpenableColumns.DISPLAY_NAME}, null, null, null)) {
   576→                    if (cursor != null && cursor.moveToFirst()) {
   577→                        int idx = cursor.getColumnIndex(OpenableColumns.DISPLAY_NAME);
   578→                        if (idx >= 0) {
   579→                            String name = cursor.getString(idx);
   580→                            if (!TextUtils.isEmpty(name)) return name;
   581→                        }
   582→                    }
   583→                }
   584→            }
   585→            if (uri != null && uri.getPath() != null) {
   586→                String path = uri.getPath();
   587→                int slash = path.lastIndexOf('/');
   588→                if (slash >= 0) return path.substring(slash + 1);
   589→            }
   590→        } catch (Exception e) {
   591→            Log.e(TAG, "Error getting file name", e);
   592→        }
   593→        return "file_" + System.currentTimeMillis();
   594→    }

   596→    private String getFolderName(Uri uri) {
   597→        try {
   598→            String lastPath = uri.getLastPathSegment();
   599→            if (lastPath != null) {
   600→                int colonIdx = lastPath.lastIndexOf(':');
   601→                if (colonIdx >= 0) lastPath = lastPath.substring(colonIdx + 1);
   602→                if (!TextUtils.isEmpty(lastPath)) return lastPath.replace("/", "_");
   603→            }
   604→        } catch (Exception e) {
   605→            Log.e(TAG, "Error getting folder name", e);
   606→        }
   607→        return "shared_folder_" + System.currentTimeMillis();
   608→    }

   610→    private void copyUriToFile(Uri uri, File destFile) throws Exception {
   611→        InputStream in = getContentResolver().openInputStream(uri);
   612→        if (in == null) throw new Exception("Cannot open file");
   613→        try (OutputStream out = new FileOutputStream(destFile)) {
   614→            byte[] buf = new byte[8192];
   615→            int len;
   616→            while ((len = in.read(buf)) > 0) out.write(buf, 0, len);
   617→        } finally {
   618→            in.close();
   619→        }
   620→    }

   622→    private void copyFolderFromUri(Uri treeUri, File destFolder) throws Exception {
   623→        String treeDocId = DocumentsContract.getTreeDocumentId(treeUri);
   624→        Uri childrenUri = DocumentsContract.buildChildDocumentsUriUsingTree(treeUri, treeDocId);
   625→
   626→        try (android.database.Cursor cursor = getContentResolver().query(childrenUri,
   627→                new String[]{
   628→                    DocumentsContract.Document.COLUMN_DOCUMENT_ID,
   629→                    DocumentsContract.Document.COLUMN_DISPLAY_NAME,
   630→                    DocumentsContract.Document.COLUMN_MIME_TYPE
   631→                }, null, null, null)) {
   632→
   633→            if (cursor != null) {
   634→                while (cursor.moveToNext()) {
   635→                    String docId = cursor.getString(0);
   636→                    String name = cursor.getString(1);
   637→                    String mimeType = cursor.getString(2);
   638→                    Uri docUri = DocumentsContract.buildDocumentUriUsingTree(treeUri, docId);
   639→                    File outFile = new File(destFolder, name);
   640→
   641→                    if (DocumentsContract.Document.MIME_TYPE_DIR.equals(mimeType)) {
   642→                        outFile.mkdirs();
   643→                        copyFolderFromUri(docUri, outFile);
   644→                    } else {
   645→                        copyUriToFile(docUri, outFile);
   646→                    }
   647→                }
   648→            }
   649→        }
   650→    }
